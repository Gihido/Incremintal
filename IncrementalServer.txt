local RunService = game:GetService("RunService")
local ServerScriptService = game:GetService("ServerScriptService")

if not RunService:IsServer() then
	warn("IncrementalServer must run as a Script on the server")
	return
end

local ServicesFolder = ServerScriptService:WaitForChild("IncrementalModules")

local GameState = require(ServicesFolder:WaitForChild("GameState"))
local CurrencyService = require(ServicesFolder:WaitForChild("CurrencyService"))
local RuneService = require(ServicesFolder:WaitForChild("RuneService"))
local PassiveService = require(ServicesFolder:WaitForChild("PassiveService"))
local UpgradeService = require(ServicesFolder:WaitForChild("UpgradeService"))
local RebirthService = require(ServicesFolder:WaitForChild("RebirthService"))
local SaveService = require(ServicesFolder:WaitForChild("SaveService"))
local AdminService = require(ServicesFolder:WaitForChild("AdminService"))

local Bootstrap = {}

function Bootstrap.buildContext()
	local context = { services = {} }

	context.services.GameState = GameState.new(context)
	context.services.CurrencyService = CurrencyService.new(context)
	context.services.PassiveService = PassiveService.new(context)
	context.services.UpgradeService = UpgradeService.new(context)
	context.services.RebirthService = RebirthService.new(context)
	context.services.RuneService = RuneService.new(context)
	context.services.SaveService = SaveService.new(context)
	context.services.AdminService = AdminService.new(context)

	return context
end

function Bootstrap.initServices(context)
	local order = {"GameState", "CurrencyService", "PassiveService", "UpgradeService", "RebirthService", "RuneService", "SaveService", "AdminService"}
	for _, name in ipairs(order) do
		local service = context.services[name]
		if service and service.init then
			service:init()
		end
	end
end

function Bootstrap.startServices(context)
	for _, service in pairs(context.services) do
		if service.start then
			service:start()
		end
	end
end

local context = Bootstrap.buildContext()
Bootstrap.initServices(context)
Bootstrap.startServices(context)
