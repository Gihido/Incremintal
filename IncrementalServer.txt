local HAY_UPGRADE_KEYS = {"HayAmount", "HayMultiplier", "HayCooldown"}

local function normalizeHaySaveData(savedHay)
	local hayTable = type(savedHay) == "table" and savedHay or {}
	hayTable.HayUpgrades = type(hayTable.HayUpgrades) == "table" and hayTable.HayUpgrades or {}
	local nestedUpgrades = hayTable.HayUpgrades
	local normalizedLevels = {}
	for _, key in ipairs(HAY_UPGRADE_KEYS) do
		local levelName = HAY_UPGRADES[key].levelName
		normalizedLevels[levelName] = tonumber(hayTable[levelName]) or tonumber(nestedUpgrades[key]) or 0
	end
	nestedUpgrades.HayAmount = normalizedLevels.HayAmountLevel or 0
	nestedUpgrades.HayMultiplier = normalizedLevels.HayMultiplierLevel or 0
	nestedUpgrades.HayCooldown = normalizedLevels.HayCooldownLevel or 0

	return {
		HayCurrency = tonumber(hayTable.HayCurrency) or 0,
		HayAmountLevel = normalizedLevels.HayAmountLevel or 0,
		HayMultiplierLevel = normalizedLevels.HayMultiplierLevel or 0,
		HayCooldownLevel = normalizedLevels.HayCooldownLevel or 0,
		HayAmountCost = tonumber(hayTable.HayAmountCost),
		HayMultiplierCost = tonumber(hayTable.HayMultiplierCost),
		HayCooldownCost = tonumber(hayTable.HayCooldownCost),
		HayUpgrades = {
			HayAmount = nestedUpgrades.HayAmount,
			HayMultiplier = nestedUpgrades.HayMultiplier,
			HayCooldown = nestedUpgrades.HayCooldown,
		},
	}
end

function packGlobalAdminEvents()
	local nowClock = os.clock()
	local nowUnix = os.time()
	local packed = {}
	for eventId, eventState in pairs(activeAdminEvents) do
		local endsAt = tonumber(eventState and eventState.endsAt) or 0
		if endsAt > nowClock then
			packed[eventId] = nowUnix + math.max(0, math.floor(endsAt - nowClock))
		end
	end
	return packed
end

function applyGlobalAdminEvents(events, skipBroadcast)
	if type(events) ~= "table" then return end
	local nowClock = os.clock()
	local nowUnix = os.time()
	local changed = false
	for eventId, unixEnd in pairs(events) do
		if ACTIVE_SERVER_EVENTS[eventId] then
			local unixTs = tonumber(unixEnd) or 0
			local clockEnd = nowClock + math.max(0, unixTs - nowUnix)
			local current = activeAdminEvents[eventId] and tonumber(activeAdminEvents[eventId].endsAt) or 0
			if clockEnd > current then
				activeAdminEvents[eventId] = {endsAt = clockEnd}
				changed = true
			end
		end
	end
	if changed and not skipBroadcast then
		broadcastAdminEventState()
	end
	return changed
end

function publishGlobalAdminEvents()
	local payload = {jobId = game.JobId, events = packGlobalAdminEvents()}
	pcall(function()
		globalEventStore:SetAsync("ACTIVE", payload.events)

function chooseRuneType(luckLevel)
	local luckPoints = math.max(0, luckLevel or 0) * 10
	local luckFactor = 1 + (luckPoints * 0.01)
	local weighted = {}
	local total = 0
	for runeId, runeDef in pairs(RUNE_TYPES) do
		local weight = safeNumber(runeDef.weight, 0)
		if runeId ~= "Grass" then
			weight *= safeNumber(luckFactor, 1)
		end
		total += safeNumber(weight, 0)
		weighted[runeId] = weight
	end
	local roll = math.random() * total
	local running = 0
	for runeId, weight in pairs(weighted) do
		running += weight
		if roll <= running then
			return runeId
		end
	end
	return "Grass"
end

function getRuneBonusMultipliers(player)
	local state = getRuneState(player)

	for runeId, runeDef in pairs(RUNE_TYPES) do
		local count = 0
		if state.counts then
			count = tonumber(state.counts[runeId]) or 0
		end
		local stacks = math.clamp(count, 0, tonumber(runeDef.capStacks) or 999999)
		coinMul += (runeDef.coinBonus or 0) * stacks
		woodMul += (runeDef.woodBonus or 0) * stacks
		paperMul += (runeDef.paperBonus or 0) * stacks
		xpMul += (runeDef.xpBonus or 0) * stacks
	end
	return coinMul, woodMul, paperMul, xpMul
end

local function getRuneSpeedOverflowBulk(speedStat)
	local interval = 1 / math.max(1, (1 + (math.max(0, speedStat) * 0.5)))
	if interval >= 0.01 then
		return 0
	end
	local overflowFactor = math.min(250, 0.01 / math.max(0.000001, interval))
	return math.max(0, math.floor((overflowFactor - 1) * 2 + 0.5))
end

local function getEffectiveRuneStats(player)
	local runeUpgrades = getRuneUpgradeFolder(player)
	local xpUpgrades = getXPUpgradesFolder(player)
	local runeState = getRuneState(player)
	if not runeUpgrades then
		return 1, 1, 1
	end
	local luck = 1 + runeUpgrades.RuneLuckLevel.Value + math.max(0, tonumber(xpUpgrades and xpUpgrades:FindFirstChild("RuneLuckXPLevel") and xpUpgrades.RuneLuckXPLevel.Value or 0))
	local speed = 1 + runeUpgrades.RuneSpeedLevel.Value + math.max(0, tonumber(xpUpgrades and xpUpgrades:FindFirstChild("RuneSpeedXPLevel") and xpUpgrades.RuneSpeedXPLevel.Value or 0))
	local bulk = 1 + runeUpgrades.RuneBulkLevel.Value + math.max(0, tonumber(xpUpgrades and xpUpgrades:FindFirstChild("RuneBulkXPLevel") and xpUpgrades.RuneBulkXPLevel.Value or 0))
	local _, _, _, _, runeLuckMul, runeSpeedMul, runeBulkAdd = getRuneBonusMultipliers(player)
	runeLuckMul = safeNumber(runeLuckMul, 1)
	runeSpeedMul = safeNumber(runeSpeedMul, 1)
	runeBulkAdd = safeNumber(runeBulkAdd, 0)
	luck = luck * runeLuckMul
	speed = speed * runeSpeedMul
	bulk = bulk + runeBulkAdd
	local passiveLuckMul, passiveSpeedMul, passiveBulkMul = getPassiveSpecialBoosts(player)
	passiveLuckMul = safeNumber(passiveLuckMul, 1)
	passiveSpeedMul = safeNumber(passiveSpeedMul, 1)
	passiveBulkMul = safeNumber(passiveBulkMul, 1)
	luck = math.floor(luck * passiveLuckMul + 0.5)
	speed = math.floor(speed * passiveSpeedMul + 0.5)
	bulk = math.floor(bulk * passiveBulkMul + 0.5)
	local eventLuckMul, eventSpeedMul, eventBulkMul, eventBulkAdd = getServerRuneEventBoosts()
	eventLuckMul = safeNumber(eventLuckMul, 1)
	eventSpeedMul = safeNumber(eventSpeedMul, 1)
	eventBulkMul = safeNumber(eventBulkMul, 1)
	eventBulkAdd = safeNumber(eventBulkAdd, 0)
	luck = math.floor(luck * eventLuckMul + 0.5)
	speed = math.floor(speed * eventSpeedMul + 0.5)
	bulk = math.floor(bulk * eventBulkMul + 0.5)
	bulk += eventBulkAdd
	local rebirth = getRebirthFolder(player)
	if rebirth and rebirth.Count.Value >= 7 then
		luck = math.floor(luck * 1.5 + 0.5)
	end
	bulk += getRuneSpeedOverflowBulk(speed)
	return luck, speed, bulk
end

function getRuneBonusMultipliers(player)
	local state = getRuneState(player)
	local coinMul, woodMul, paperMul, xpMul = 1, 1, 1, 1
	local runeLuckMul, runeSpeedMul, runeBulkAdd, passiveLuckMul = 1, 1, 0, 1
	for runeId, runeDef in pairs(RUNE_TYPES) do
		local count = 0
		if state.counts then
			count = safeNumber(state.counts[runeId], 0)
		end
		local stacks = math.clamp(safeNumber(count, 0), 0, safeNumber(runeDef.capStacks, DEFAULT_RUNE_CAP_STACKS))
		coinMul += safeNumber(runeDef.coinBonus, 0) * stacks
		woodMul += safeNumber(runeDef.woodBonus, 0) * stacks
		paperMul += safeNumber(runeDef.paperBonus, 0) * stacks
		xpMul += safeNumber(runeDef.xpBonus, 0) * stacks
		runeLuckMul += safeNumber(runeDef.runeLuckBonus, 0) * stacks
		runeSpeedMul += safeNumber(runeDef.runeSpeedBonus, 0) * stacks
		runeBulkAdd += safeNumber(runeDef.runeBulkBonus, 0) * stacks
		passiveLuckMul += safeNumber(runeDef.passiveLuckBonus, 0) * stacks
	end
	local forest = state and state.forestCounts or {}
	local palka = math.min(15, safeNumber(forest.Palka, 0))
	local obrub = math.min(56, safeNumber(forest.ObrublennyKonec, 0))
	local vetv = math.min(40, safeNumber(forest.VetvDereva, 0))
	local brevno = math.min(20, safeNumber(forest.Brevno, 0))
	local poleno = math.min(40, safeNumber(forest.Poleno, 0))
	local obgPen = math.min(20, safeNumber(forest.ObgorevshiyPen, 0))
	coinMul += (palka * 0.1) + (obrub * 0.25)
	xpMul += vetv * 0.1
	runeLuckMul += brevno * 0.1
	runeSpeedMul += brevno * 0.1
	runeBulkAdd += math.min(5, brevno)
	paperMul += poleno * 0.1
	woodMul += poleno * 0.1
	passiveLuckMul += obgPen * 0.1
	return coinMul, woodMul, paperMul, xpMul, runeLuckMul, runeSpeedMul, runeBulkAdd, passiveLuckMul
end

		runeBulkAdd += safeNumber(runeDef.runeBulkBonus, 0) * stacks
		passiveLuckMul += safeNumber(runeDef.passiveLuckBonus, 0) * stacks
	end
	local forest = state and state.forestCounts or {}
	local palka = math.min(15, safeNumber(forest.Palka, 0))
	local obrub = math.min(56, safeNumber(forest.ObrublennyKonec, 0))
	local vetv = math.min(40, safeNumber(forest.VetvDereva, 0))
	local brevno = math.min(20, safeNumber(forest.Brevno, 0))
	local poleno = math.min(40, safeNumber(forest.Poleno, 0))
	local obgPen = math.min(20, safeNumber(forest.ObgorevshiyPen, 0))
	coinMul += (palka * 0.1) + (obrub * 0.25)
	xpMul += vetv * 0.1
	runeLuckMul += brevno * 0.1
	runeSpeedMul += brevno * 0.1
	runeBulkAdd += math.min(5, brevno)
	paperMul += poleno * 0.1
	woodMul += poleno * 0.1
	passiveLuckMul += obgPen * 0.1
	return coinMul, woodMul, paperMul, xpMul, runeLuckMul, runeSpeedMul, runeBulkAdd, passiveLuckMul
end

function chooseRuneType(luckLevel)
	local luckFactor = 1 + (math.max(0, luckLevel or 0) * 0.12)
	local weighted = {}
	local total = 0
	for runeId, runeDef in pairs(RUNE_TYPES) do
		local weight = safeNumber(runeDef.weight, 0)
		if runeId ~= "Grass" then
			weight *= safeNumber(luckFactor, 1)
		end
		total += safeNumber(weight, 0)
		weighted[runeId] = weight
	end
	local roll = math.random() * total
	local running = 0
	for runeId, weight in pairs(weighted) do
		running += weight
		if roll <= running then
			return runeId
		end
	end
	return "Grass"
end

function getRuneBonusMultipliers(player)

	local coinMul, woodMul, paperMul, xpMul = 1, 1, 1, 1

		local count = 0
		if state.counts then
			count = safeNumber(state.counts[runeId], 0)
		end
		local stacks = math.clamp(safeNumber(count, 0), 0, safeNumber(runeDef.capStacks, DEFAULT_RUNE_CAP_STACKS))
		coinMul += safeNumber(runeDef.coinBonus, 0) * stacks
		woodMul += safeNumber(runeDef.woodBonus, 0) * stacks
		paperMul += safeNumber(runeDef.paperBonus, 0) * stacks
		xpMul += safeNumber(runeDef.xpBonus, 0) * stacks
		runeLuckMul += safeNumber(runeDef.runeLuckBonus, 0) * stacks
		runeSpeedMul += safeNumber(runeDef.runeSpeedBonus, 0) * stacks
		runeBulkAdd += safeNumber(runeDef.runeBulkBonus, 0) * stacks
		passiveLuckMul += safeNumber(runeDef.passiveLuckBonus, 0) * stacks
	end
	local forest = state and state.forestCounts or {}
	local palka = math.min(15, safeNumber(forest.Palka, 0))
	local obrub = math.min(56, safeNumber(forest.ObrublennyKonec, 0))
	local vetv = math.min(40, safeNumber(forest.VetvDereva, 0))
	local brevno = math.min(20, safeNumber(forest.Brevno, 0))
	local poleno = math.min(40, safeNumber(forest.Poleno, 0))
	local obgPen = math.min(20, safeNumber(forest.ObgorevshiyPen, 0))
	coinMul += (palka * 0.1) + (obrub * 0.25)
	xpMul += vetv * 0.1
	runeLuckMul += brevno * 0.1
	runeSpeedMul += brevno * 0.1
	runeBulkAdd += math.min(5, brevno)
	paperMul += poleno * 0.1
	woodMul += poleno * 0.1
	passiveLuckMul += obgPen * 0.1
	return coinMul, woodMul, paperMul, xpMul, runeLuckMul, runeSpeedMul, runeBulkAdd, passiveLuckMul
end

local function chooseRandomPassiveId(player)
	local totalWeight = 0
	local passiveLuckMul = 1
	if player then
		_, _, _, _, _, _, _, passiveLuckMul = getRuneBonusMultipliers(player)
	end
	for _, passiveDef in pairs(PASSIVE_DEFS) do
		totalWeight += safeNumber(passiveDef.weight, 0) * safeNumber(passiveLuckMul, 1)
	end

	local roll = math.random() * totalWeight
	local running = 0

	for passiveId, passiveDef in pairs(PASSIVE_DEFS) do
		running += safeNumber(passiveDef.weight, 0) * safeNumber(passiveLuckMul, 1)
		if roll <= running then
			return passiveId
		end
	end

	return "CoinBallA"
end

local function applyCharacterCollisionGroup(player, character)
	local groupName = getPlayerCollisionGroup(player)

	local function applyPart(instance)
		if instance:IsA("BasePart") then
			instance.CollisionGroup = groupName
		end
	end

	applyPart(character)
	for _, descendant in ipairs(character:GetDescendants()) do
		applyPart(descendant)
	end

	if characterConnections[player] then
		characterConnections[player]:Disconnect()
	end

	characterConnections[player] = character.DescendantAdded:Connect(applyPart)
end

local function getRebirthCoinMultiplier(count)
	local c = math.max(0, math.floor(tonumber(count) or 0))
	if c == 0 then return 1 end
	if c == 1 then return 2 end
	if c == 2 then return 10 end
	if c == 3 then return 100 end

	local mul = 1
	for i = 1, c do
		local stage = REBIRTH_STAGES[i]
		if stage then
			mul *= (tonumber(stage.coinMultiplier) or 1)
		end
	end
	return mul
end

local function syncRebirthCoinMultiplierBonus(player)
	local rebirth = getRebirthFolder(player)
	if not rebirth then return 1 end
	local count = tonumber(rebirth.Count.Value) or 0
	local rebirthMul = getRebirthCoinMultiplier(count)
	if rebirth.CoinMultiplierBonus.Value ~= rebirthMul then
		rebirth.CoinMultiplierBonus.Value = rebirthMul
	end
	return rebirthMul
end

local function runRebirthCoinMultiplierSanityCheck()
	if not DEV_REBIRTH_LOGS then
		return
	end
	for rebirthCount = 0, 3 do
		local baseCoinGain = 1
		local finalCoinGain = baseCoinGain * getRebirthCoinMultiplier(rebirthCount)
		warn(string.format("[RebirthSanity] rebirth=%d base=%s final=%s", rebirthCount, tostring(baseCoinGain), tostring(finalCoinGain)))
	end
end

local function recalculateRebirthStats(player)
	local rebirth = getRebirthFolder(player)
	if not rebirth then
		return
	end

	local count = rebirth.Count.Value
	local totalCoinMultiplier = 1
	local totalSpawnBonus = 0
	local totalWoodMultiplier = 1

	local secondUnlocked = false
	local thirdUnlocked = false
	local fourthUnlocked = false
	local fifthUnlocked = false
	local sixthUnlocked = false

	for i = 1, count do
		local stage = REBIRTH_STAGES[i]
		if stage then
			totalCoinMultiplier *= (stage.coinMultiplier or 1)
			totalSpawnBonus += (stage.spawnBonus or 0)
			totalWoodMultiplier *= (stage.woodMultiplier or 1)

			if stage.unlockTree then
				secondUnlocked = true
			end
			if stage.unlockThirdUpgrades then
				thirdUnlocked = true
			end
			if stage.unlockFourthSystems then
				fourthUnlocked = true
			end
				if stage.unlockFifthSystems then
					fifthUnlocked = true
				end
				if stage.unlockSixthSystems then
					sixthUnlocked = true
				end
			end
		end


	rebirth.CoinMultiplierBonus.Value = syncRebirthCoinMultiplierBonus(player)
	rebirth.SpawnSpeedBonus.Value = totalSpawnBonus
	rebirth.WoodMultiplierBonus.Value = totalWoodMultiplier
	rebirth.SecondAreaUnlocked.Value = secondUnlocked
	rebirth.ThirdUpgradeUnlocked.Value = thirdUnlocked
	rebirth.FourthSystemsUnlocked.Value = fourthUnlocked
	rebirth.FifthSystemsUnlocked.Value = fifthUnlocked
	rebirth.SixthSystemsUnlocked.Value = sixthUnlocked

	local nextStage = REBIRTH_STAGES[count + 1]
	if nextStage then
		rebirth.NextCost.Value = nextStage.cost
		rebirth.NextCurrency.Value = nextStage.currency
	else
		rebirth.NextCost.Value = 0
		rebirth.NextCurrency.Value = "None"
	end

	if player.Character then
		applyCharacterCollisionGroup(player, player.Character)
	end
end

local function computeGoalWoodFactorFromCoins(coinAmount)
	local threshold = 10000
	local goalsReached = 0
	local factor = 1

	while factor < 10 and coinAmount >= threshold do
		goalsReached += 1
		factor = math.min(1 + (goalsReached * 0.1), 10)
		threshold = math.floor(threshold * 1.25 + 0.5)
	end

	return factor
end

local function calculateFinalCoinGain(player, baseAmount)
	local coinUpgrades = getCoinUpgradesFolder(player)
	local woodUpgrades = getWoodUpgradesFolder(player)
	local rebirth = getRebirthFolder(player)
	local passiveCoin, _, passiveGlobalMultiplier = getPassiveMultipliers(player)
	local runeCoinMultiplier = select(1, getRuneBonusMultipliers(player))

	local baseCoinGain = math.max(0, tonumber(baseAmount) or 1)
	local coinValueLevel = coinUpgrades and coinUpgrades:FindFirstChild("CoinValueLevel") and coinUpgrades.CoinValueLevel.Value or 0
	local multiplierLevel = coinUpgrades and coinUpgrades:FindFirstChild("MultiplierLevel") and coinUpgrades.MultiplierLevel.Value or 0
	local upgradedCoinGain = (baseCoinGain + coinValueLevel) * (1 + (multiplierLevel * 0.1))
	local rebirthCoinMul = rebirth and syncRebirthCoinMultiplierBonus(player) or 1
	local afterRebirthGain = upgradedCoinGain * rebirthCoinMul
	local woodCoinBoostLevel = woodUpgrades and woodUpgrades:FindFirstChild("CoinBoostLevel") and woodUpgrades.CoinBoostLevel.Value or 0
	local woodCoinBoostMultiplier = 1.1 ^ woodCoinBoostLevel
	local xpUpgrades = getXPUpgradesFolder(player)
	local xpMul = 1 + ((xpUpgrades and xpUpgrades:FindFirstChild("CoinXPLevel") and xpUpgrades.CoinXPLevel.Value or 0) * 0.2)
	local eventCoinMul = getServerEventMultipliers()
	local finalCoinGain = afterRebirthGain * woodCoinBoostMultiplier * passiveCoin * passiveGlobalMultiplier * runeCoinMultiplier * xpMul * eventCoinMul
	local floorGain = baseCoinGain * rebirthCoinMul
	return roundToTenth(math.max(finalCoinGain, floorGain))
end

local function getCoinRespawnDelay(player)
	local coinUpgrades = getCoinUpgradesFolder(player)
	local rebirth = getRebirthFolder(player)

	if not coinUpgrades or not rebirth then
		return BASE_COIN_RESPAWN
	end

	local value = BASE_COIN_RESPAWN
	value -= coinUpgrades.SpawnSpeedLevel.Value * 0.5
	value -= rebirth.SpawnSpeedBonus.Value
	local _, _, _, passiveSpeedMultiplier = getPassiveMultipliers(player)
	value /= math.max(1, passiveSpeedMultiplier)

	if value < MIN_COIN_RESPAWN then
		value = MIN_COIN_RESPAWN
	end

	return roundToTenth(value)
end

local function getWoodReward(player)
	local coinUpgrades = getCoinUpgradesFolder(player)
	local woodUpgrades = getWoodUpgradesFolder(player)
	local rebirth = getRebirthFolder(player)
	local coins = getCoinsObject(player)

	if not coinUpgrades or not woodUpgrades or not rebirth or not coins then
		return 1
	end

	local _, passiveWoodMultiplier, passiveGlobalMultiplier = getPassiveMultipliers(player)
	local _, runeWoodMultiplier = getRuneBonusMultipliers(player)

	local baseValue = 1 + woodUpgrades.WoodValueLevel.Value
	local woodMultiplier = 1 + woodUpgrades.WoodMultiplierLevel.Value + (coinUpgrades.WoodBoostLevel.Value * 0.25)

	local goalFactor = 1
	if rebirth.FourthSystemsUnlocked.Value then
		goalFactor = computeGoalWoodFactorFromCoins(coins.Value)
	end

	local xpUpgrades = getXPUpgradesFolder(player)
	local xpMul = 1 + ((xpUpgrades and xpUpgrades:FindFirstChild("WoodXPLevel") and xpUpgrades.WoodXPLevel.Value or 0) * 0.2)
	local _, eventWoodMul = getServerEventMultipliers()
	return roundToTenth(baseValue * woodMultiplier * goalFactor * rebirth.WoodMultiplierBonus.Value * passiveWoodMultiplier * passiveGlobalMultiplier * runeWoodMultiplier * xpMul * eventWoodMul)
end

local function getWoodCooldown(player)
	local woodUpgrades = getWoodUpgradesFolder(player)
	if not woodUpgrades then
		return BASE_WOOD_COOLDOWN
	end

	local value = BASE_WOOD_COOLDOWN - (woodUpgrades.WoodSpeedLevel.Value * WOOD_COOLDOWN_STEP)
	local _, _, _, passiveSpeedMultiplier = getPassiveMultipliers(player)
	value /= math.max(1, passiveSpeedMultiplier)
	if value < MIN_WOOD_COOLDOWN then
		value = MIN_WOOD_COOLDOWN
	end

	return roundToTenth(value)
end

local function getHayReward(player)
	local hayUpgrades = getHayUpgradesFolder(player)
	local rebirth = getRebirthFolder(player)
	if not hayUpgrades or not rebirth or (tonumber(rebirth.Count.Value) or 0) < 7 then
		return 0
	end
	local amountLevel = hayUpgrades.HayAmountLevel.Value
	local multiplierLevel = hayUpgrades.HayMultiplierLevel.Value
	local base = 1 + amountLevel
	local multiplier = 1 + (multiplierLevel * 0.5)
	return roundToTenth(base * multiplier)
end

local function getHayCooldown(player)
	local hayUpgrades = getHayUpgradesFolder(player)
	local rebirth = getRebirthFolder(player)
	if not hayUpgrades or not rebirth or (tonumber(rebirth.Count.Value) or 0) < 7 then
		return BASE_HAY_COOLDOWN
	end
	local level = hayUpgrades.HayCooldownLevel.Value
	return math.max(MIN_HAY_COOLDOWN, roundToTenth(BASE_HAY_COOLDOWN - (level * 0.1)))
end

local function getPaperProductionAmount(player)
	local paperUpgrades = getPaperUpgradesFolder(player)
	if not paperUpgrades then
		return 1
	end

	local baseValue = 1 + paperUpgrades.PaperValueLevel.Value
	local multiplier = 1 + (paperUpgrades.PaperMultiplierLevel.Value * 0.1)
	local _, _, runePaperMultiplier = getRuneBonusMultipliers(player)

	local xpUpgrades = getXPUpgradesFolder(player)
	local xpMul = 1 + ((xpUpgrades and xpUpgrades:FindFirstChild("PaperXPLevel") and xpUpgrades.PaperXPLevel.Value or 0) * 0.2)
	local _, _, eventPaperMul = getServerEventMultipliers()
	return roundToTenth(baseValue * multiplier * runePaperMultiplier * xpMul * eventPaperMul)
end

local function getPaperCycleTime(player)
	local paperUpgrades = getPaperUpgradesFolder(player)
	if not paperUpgrades then
		return BASE_PAPER_PRODUCTION_TIME
	end

	local value = BASE_PAPER_PRODUCTION_TIME - (paperUpgrades.PaperSpeedLevel.Value * PAPER_SPEED_STEP)
	if value < MIN_PAPER_PRODUCTION_TIME then
		value = MIN_PAPER_PRODUCTION_TIME
	end

	return roundToTenth(value)
end

local function addCurrency(currencyObject, amount)
	if currencyObject then
		currencyObject.Value = roundToTenth(currencyObject.Value + amount)
	end
end

local function spendCurrency(currencyObject, amount)
	if not currencyObject then
		return false
	end

	if currencyObject.Value < amount then
		return false
	end

	currencyObject.Value = roundToTenth(currencyObject.Value - amount)
	return true
end

local function getCurrencyObjectForName(player, currencyName)
	if currencyName == "Coins" then
		return getCoinsObject(player)
	elseif currencyName == "Wood" then
		return getWoodCurrencyObject(player)
	elseif currencyName == "Paper" then
		return getPaperCurrencyObject(player)
	elseif currencyName == "XP" then
		return getXPCurrencyObject(player)
	elseif currencyName == "Hay" then
		return getHayCurrencyObject(player)
	end
	return nil
end

local function tryBuyUpgrade(player, upgradeKey, configTable, folder, currencyObject, currentRebirthCount)
	local config = configTable[upgradeKey]
	if not config then
		return "INVALID"
	end

	if config.requiredRebirth and currentRebirthCount < config.requiredRebirth then
		return "LOCKED"
	end

	local levelObject = folder:FindFirstChild(config.levelName)
	local costObject = folder:FindFirstChild(config.costName)

	if not levelObject or not costObject then
		return "INVALID"
	end

	if levelObject.Value >= config.maxLevel then
		costObject.Value = 0
		updateUpgradeActiveFlags(folder, configTable)
		return "AT_MAX"
	end

	local cost = costObject.Value
	if cost <= 0 then
		return "AT_MAX"
	end

	if not spendCurrency(currencyObject, cost) then
		return "NO_MONEY"
	end

	levelObject.Value += 1

	if levelObject.Value >= config.maxLevel then
		costObject.Value = 0
	elseif config.fixedCost then
		costObject.Value = config.startCost
	else
		costObject.Value = math.ceil(cost * config.priceMultiplier)
	end

	updateUpgradeActiveFlags(folder, configTable)
	markDirty(player)

	return "BOUGHT"
end

local function handleUpgradePurchase(player, upgradeFamily, upgradeKey, mode)
	local rebirth = getRebirthFolder(player)
	if not rebirth then
		return
	end

	local configTable, folder, currencyObject, notEnoughText, lockedText

	if upgradeFamily == "Coin" then
		configTable = COIN_UPGRADES
		folder = getCoinUpgradesFolder(player)
		currencyObject = getCoinsObject(player)
		notEnoughText = "Не хватает монет"
		lockedText = "Откроется на 3-м перерождении"
	elseif upgradeFamily == "Wood" then
		if not rebirth.SecondAreaUnlocked.Value then
			fireSimple(player, "Дерево откроется на 2-м перерождении")
			return
		end
		configTable = WOOD_UPGRADES
		folder = getWoodUpgradesFolder(player)
		currencyObject = getWoodCurrencyObject(player)
		notEnoughText = "Не хватает дерева"
		lockedText = "Откроется на 3-м перерождении"
	elseif upgradeFamily == "Paper" then
		if not rebirth.FourthSystemsUnlocked.Value then
			fireSimple(player, "Откроется на 4-м перерождении")
			return
		end
		configTable = PAPER_UPGRADES
		folder = getPaperUpgradesFolder(player)
		currencyObject = getPaperCurrencyObject(player)
		notEnoughText = "Не хватает бумаги"
		lockedText = "Откроется на 4-м перерождении"
	elseif upgradeFamily == "XP" then
		if not rebirth.SecondAreaUnlocked.Value then
			fireSimple(player, "XP откроется на 2-м перерождении")
			return
		end
		configTable = XP_UPGRADES
		folder = getXPUpgradesFolder(player)
		currencyObject = getXPCurrencyObject(player)
		notEnoughText = "Не хватает XP"
		lockedText = "Откроется на 2-м перерождении"
	elseif upgradeFamily == "Hay" then
		if (tonumber(rebirth.Count.Value) or 0) < 7 then
			fireSimple(player, "Hay откроется на 7-м перерождении")
			return
		end
		configTable = HAY_UPGRADES
		folder = getHayUpgradesFolder(player)
		currencyObject = getHayCurrencyObject(player)
		notEnoughText = "Не хватает Hay"
		lockedText = "Откроется на 7-м перерождении"
	else
		return
	end

	if not folder or not currencyObject then
		return
	end

	if mode == "Max" then
		local bought = 0

		for _ = 1, 500 do
			local result = tryBuyUpgrade(player, upgradeKey, configTable, folder, currencyObject, rebirth.Count.Value)
			if result == "BOUGHT" then
				bought += 1
			else
				if bought > 0 then
					fireSimple(player, "Куплено уровней: " .. bought)
				elseif result == "AT_MAX" then
					fireSimple(player, "Улучшение уже в MAX")
				elseif result == "LOCKED" then
					fireSimple(player, lockedText)
				else
					fireSimple(player, notEnoughText)
				end
				break
			end
		end
	else
		local result = tryBuyUpgrade(player, upgradeKey, configTable, folder, currencyObject, rebirth.Count.Value)
		if result == "BOUGHT" then
			fireSimple(player, "Улучшение куплено")
		elseif result == "AT_MAX" then
			fireSimple(player, "Улучшение уже в MAX")
		elseif result == "LOCKED" then
			fireSimple(player, lockedText)
		else
			fireSimple(player, notEnoughText)
		end
	end
end

local function tryPurchaseRebirth(player)
	local rebirth = getRebirthFolder(player)
	if not rebirth then
		return
	end

	local currentCount = rebirth.Count.Value
	if currentCount >= MAX_REBIRTH_COUNT then
		fireSimple(player, "Перерождение уже в MAX")
		return
	end

	local nextStage = REBIRTH_STAGES[currentCount + 1]
	if not nextStage then
		fireSimple(player, "Перерождение уже в MAX")
		return
	end

	local currencyObject = getCurrencyObjectForName(player, nextStage.currency)
	if not currencyObject then
		return
	end

	if currencyObject.Value < nextStage.cost then
		if nextStage.currency == "Coins" then
			fireSimple(player, "Не хватает монет для перерождения")
		elseif nextStage.currency == "Wood" then
			fireSimple(player, "Не хватает дерева для перерождения")
		else
			fireSimple(player, "Не хватает бумаги для перерождения")
		end
		return
	end

	if not spendCurrency(currencyObject, nextStage.cost) then
		return
	end

	rebirth.Count.Value += 1
	recalculateRebirthStats(player)

	resetCoinProgress(player)
	resetWoodProgress(player)

	markDirty(player)
	fireSimple(player, "Перерождение куплено")
end

local function handleFactoryAction(player, actionName)
	local rebirth = getRebirthFolder(player)
	local woodCurrency = getWoodCurrencyObject(player)
	local paperFactory = getPaperFactoryFolder(player)

	if not rebirth or not woodCurrency or not paperFactory then
		return
	end

	if not rebirth.FourthSystemsUnlocked.Value then
		fireSimple(player, "Откроется на 4-м перерождении")
		return
	end

	if actionName == "AddFuelOnce" then
		if woodCurrency.Value < WOOD_PER_FUEL then
			fireSimple(player, "Не хватает дерева")
			return
		end

		if spendCurrency(woodCurrency, WOOD_PER_FUEL) then
			paperFactory.Fuel.Value += 1
			paperProcessAnchors[player] = os.clock()
			markDirty(player)
			fireSimple(player, "Топливо добавлено")
		end
	elseif actionName == "AddFuelMax" then
		local fuelToAdd = math.floor(woodCurrency.Value / WOOD_PER_FUEL)
		if fuelToAdd <= 0 then
			fireSimple(player, "Не хватает дерева")
			return
		end

		local totalCost = fuelToAdd * WOOD_PER_FUEL
		if spendCurrency(woodCurrency, totalCost) then
			paperFactory.Fuel.Value += fuelToAdd
			paperProcessAnchors[player] = os.clock()
			markDirty(player)
			fireSimple(player, "Добавлено топлива: " .. fuelToAdd)
		end
	end
end

local function handlePassiveAction(player, actionName, argument)
	local rebirth = getRebirthFolder(player)
	local paper = getPaperCurrencyObject(player)

	if not rebirth or not paper then
		return
	end

	if not rebirth.FifthSystemsUnlocked.Value then
		fireSimple(player, "Откроется на 5-м перерождении")
		return
	end

	local state = getPassiveState(player)

	if actionName == "Roll" then
		local now = os.clock()
		local nextRollAt = passiveRollCooldowns[player.UserId] or 0
		if now < nextRollAt then
			fireSimple(player, "Подожди, предыдущий ролл еще идет")
			return
		end
		if #state.inventory >= PASSIVE_INVENTORY_CAPACITY then
			fireSimple(player, "Инвентарь пассивов заполнен")
			return
		end

		if paper.Value < PASSIVE_ROLL_COST then
			fireSimple(player, "Не хватает бумаги")
			return
		end

		if not spendCurrency(paper, PASSIVE_ROLL_COST) then
			return
		end

		local passiveId = chooseRandomPassiveId(player)
		passiveRollCooldowns[player.UserId] = now + PASSIVE_ROLL_COOLDOWN
		table.insert(state.inventory, {
			uid = HttpService:GenerateGUID(false),
			passiveId = passiveId,
		})

		writePassiveStateValue(player)
		markDirty(player)
		firePassiveRoll(player, passiveId)
	elseif actionName == "Equip" then
		local targetUid = tostring(argument or "")
		local found = false

		for _, entry in ipairs(state.inventory) do
			if entry.uid == targetUid then
				found = true
				break
			end
		end

		if found then
			state.equippedUid = targetUid
			writePassiveStateValue(player)
			markDirty(player)
			fireSimple(player, "Пассив экипирован")
		end
	elseif actionName == "Unequip" then
		state.equippedUid = ""
		writePassiveStateValue(player)
		markDirty(player)
		fireSimple(player, "Пассив снят")
	elseif actionName == "Delete" then
		local targetUid = tostring(argument or "")

		for i = #state.inventory, 1, -1 do
			if state.inventory[i].uid == targetUid then
				table.remove(state.inventory, i)
				break
			end
		end

		if state.equippedUid == targetUid then
			state.equippedUid = ""
		end

		writePassiveStateValue(player)
		markDirty(player)
		fireSimple(player, "Пассив удалён")
	end
end

local function getRuneOpenDuration(player)
	local upgrades = getRuneUpgradeFolder(player)
	if not upgrades then
		return RUNE_OPEN_BASE_TIME
	end
	local speedLevel = upgrades.RuneSpeedLevel.Value
	local t = RUNE_OPEN_BASE_TIME - (speedLevel * 2.5)
	return math.max(5, t)
end

function getEffectiveRuneChanceDenominator(baseDenominator, runeLuck)
	local base = math.max(1, math.floor(tonumber(baseDenominator) or 1))
	local luck = math.max(1, tonumber(runeLuck) or 1)
	local effective = math.floor(base / luck + 0.5)
	return math.max(1, effective)
end

local function isCharacterOnBlock(player, block, tolerance)
	if not block then
		return false
	end
	local character = player.Character
	local humanoid = character and character:FindFirstChildOfClass("Humanoid")
	if not humanoid or humanoid.Health <= 0 then
		return false
	end
	local hrp = character and character:FindFirstChild("HumanoidRootPart")
	if not hrp then
		return false
	end
	local tol = tolerance or Vector3.new(1.2, 3.5, 1.2)
	local localPos = block.CFrame:PointToObjectSpace(hrp.Position)
	local half = block.Size * 0.5
	return math.abs(localPos.X) <= (half.X + tol.X)
		and math.abs(localPos.Z) <= (half.Z + tol.Z)
		and math.abs(localPos.Y) <= (half.Y + tol.Y)
end

local function rollFromSet(player, setName)
	local setDef = RUNE_SET_DEFS[setName]
	if not setDef then
		return nil
	end
	local state = getRuneState(player)
	state.natureCounts = state.natureCounts or {}
	state.forestCounts = state.forestCounts or {}
	local targetCounts = setName == "Forest" and state.forestCounts or state.natureCounts
	local luck, _, bulkLevel = getEffectiveRuneStats(player)
	local bulk = math.max(1, 1 + math.floor(tonumber(bulkLevel) or 0))
	local rolled = {}
	local denominators = {}
	for _, runeId in ipairs(setDef.order) do
		denominators[runeId] = getEffectiveRuneChanceDenominator(setDef.baseDenominators[runeId], luck)
	end
	for _ = 1, bulk do
		local chosen = setDef.order[1]
		for i = #setDef.order, 1, -1 do
			local runeId = setDef.order[i]
			if math.random(1, denominators[runeId]) == 1 then
				chosen = runeId
				break
			end
		end
		targetCounts[chosen] = (targetCounts[chosen] or 0) + 1
		table.insert(rolled, chosen)
	end
	state.counts = state.natureCounts
	writeRuneStateValue(player)
	markDirty(player)
	return rolled, state, denominators
end

local function getRuneRollInterval(player)
	local _, speed = getEffectiveRuneStats(player)
	local baseInterval = 1.2
	local interval = baseInterval / math.max(1, speed)
	return math.max(0.2, interval)
end

local function startRuneRolling(player, setName)
	local session = runeRollingSessions[player] or {}
	if session.active and session.setName == setName then
		return
	end
	runeRollingSessions[player] = {
		active = true,
		setName = setName,
		lastRollAt = 0,
		sessionCounts = {},
	}
	notifyEvent:FireClient(player, {kind = "rune_roll_state", active = true, system = setName})
end

local function stopRuneRolling(player, setName)
	local session = runeRollingSessions[player]
	if not session or not session.active or (setName and session.setName ~= setName) then
		return
	end
	session.active = false
	notifyEvent:FireClient(player, {
		kind = "rune_roll_state",
		active = false,
		system = session.setName,
		session = session.sessionCounts,
	})
end

local function isCharacterOnHayBlock(player)
	if not hayBlock then
		return false
	end
	local rebirth = getRebirthFolder(player)
	if not rebirth or (tonumber(rebirth.Count.Value) or 0) < 7 then
		return false
	end
	return isCharacterOnBlock(player, hayBlock, Vector3.new(0.2, 3, 0.2))
end

local function tryBuyRuneUpgrade(player, upgradeKey, silent)
	local cfg = RUNE_UPGRADES[upgradeKey]
	local upgrades = getRuneUpgradeFolder(player)
	if not cfg or not upgrades then return end
	local rebirth = getRebirthFolder(player)
	if not rebirth or not rebirth.SecondAreaUnlocked.Value then
		if not silent then fireSimple(player, "Forest Runes откроются на 2-м перерождении") end
		return
	end
	local levelObj = upgrades:FindFirstChild(cfg.levelName)
	local costObj = upgrades:FindFirstChild(cfg.costName)
	if not levelObj or not costObj then return end
	if levelObj.Value >= cfg.maxLevel then
		if not silent then fireSimple(player, "Улучшение на максимуме") end
		return
	end
	local currencyObj
	if cfg.currency == "Coins" then
		currencyObj = getCoinsObject(player)
	elseif cfg.currency == "Wood" then
		currencyObj = getWoodCurrencyObject(player)
	else
		currencyObj = getPaperCurrencyObject(player)
	end
	if not currencyObj or currencyObj.Value < costObj.Value then
		if not silent then fireSimple(player, "Недостаточно ресурса") end
		return
	end
	if not spendCurrency(currencyObj, costObj.Value) then
		return
	end
	levelObj.Value += 1
	costObj.Value = math.floor(costObj.Value * 2 + 0.5)
	if not silent then fireSimple(player, "Улучшение рун куплено") end
	markDirty(player)
end

local function tryBuyRuneUpgradeMax(player, upgradeKey)
	local safety = 0
	while safety < 200 do
		safety += 1
		local before = getRuneUpgradeFolder(player)
		local cfg = RUNE_UPGRADES[upgradeKey]
		local levelBefore = before and cfg and before:FindFirstChild(cfg.levelName) and before[cfg.levelName].Value or nil
		tryBuyRuneUpgrade(player, upgradeKey, true)
		local after = getRuneUpgradeFolder(player)
		local levelAfter = after and cfg and after:FindFirstChild(cfg.levelName) and after[cfg.levelName].Value or nil
		if levelBefore == nil or levelAfter == nil or levelAfter <= levelBefore then
			break
		end
		if levelAfter >= cfg.maxLevel then
			break
		end
	end
end

local function handleRuneAction(player, actionName)
	local rebirth = getRebirthFolder(player)
	local runeUpgrades = getRuneUpgradeFolder(player)
	local wood = getWoodCurrencyObject(player)
	if not rebirth or not runeUpgrades or not wood then return end

	if actionName == "Open" or actionName == "OpenNature" then
		fireSimple(player, "Nature Rune работает автоматически, пока стоишь на NatureRuneBlock")
		return
	elseif actionName == "OpenForest" then
		fireSimple(player, "Forest Rune работает автоматически, пока стоишь на ForestRuneBlock")
		return
	elseif not rebirth.SecondAreaUnlocked.Value then
		fireSimple(player, "Система Nature Rune откроется после 2-го перерождения")
		return
	elseif actionName == "UpgradeLuck" then
		tryBuyRuneUpgrade(player, "Luck")
	elseif actionName == "UpgradeLuckMax" then
		tryBuyRuneUpgradeMax(player, "Luck")
	elseif actionName == "UpgradeSpeed" then
		tryBuyRuneUpgrade(player, "Speed")
	elseif actionName == "UpgradeSpeedMax" then
		tryBuyRuneUpgradeMax(player, "Speed")
	elseif actionName == "UpgradeBulk" then
		tryBuyRuneUpgrade(player, "Bulk")
	elseif actionName == "UpgradeBulkMax" then
		tryBuyRuneUpgradeMax(player, "Bulk")
	end
end

local function handleXPAction(player, actionName)
	local rebirth = getRebirthFolder(player)
	local xp = getXPCurrencyObject(player)
	local xpBoost = getXPBoostFolder(player)
	if not rebirth or not xp or not xpBoost then
		return
	end
	if not rebirth.SecondAreaUnlocked.Value then
		fireSimple(player, "XP откроется на 2-м перерождении")
		return
	end
	if actionName ~= "BuyXPBoost" then
		return
	end
	local levelObj = xpBoost:FindFirstChild(XP_BOOST_CONFIG.levelName)
	local costObj = xpBoost:FindFirstChild(XP_BOOST_CONFIG.costName)
	if not levelObj or not costObj then
		return
	end
	if levelObj.Value >= XP_BOOST_CONFIG.maxLevel then
		fireSimple(player, "XP Boost уже в MAX")
		return
	end
	if not spendCurrency(xp, costObj.Value) then
		fireSimple(player, "Не хватает XP")
		return
	end
	levelObj.Value += 1
	if levelObj.Value >= XP_BOOST_CONFIG.maxLevel then
		costObj.Value = 0
	else
		costObj.Value = math.floor(costObj.Value * XP_BOOST_CONFIG.priceMultiplier + 0.5)
	end
	markDirty(player)
	fireSimple(player, "XP Boost улучшен")
end

local function getRandomCoinCFrame()
	local zoneSize = zonePart.Size
	local coinSize = coinTemplate.Size

	local halfX = math.max((zoneSize.X / 2) - (coinSize.X / 2), 0)
	local halfZ = math.max((zoneSize.Z / 2) - (coinSize.Z / 2), 0)

	local randomX = (math.random() * 2 - 1) * halfX
	local randomZ = (math.random() * 2 - 1) * halfZ
	local yOffset = (zoneSize.Y / 2) + (coinSize.Y / 2) + 0.05

	local localPosition = Vector3.new(randomX, yOffset, randomZ)
	local worldPosition = zonePart.CFrame:PointToWorldSpace(localPosition)

	return CFrame.new(worldPosition)
end

local function applyCoinVisuals(coin)
	if not coin:FindFirstChild("CoinLight") then
		local light = Instance.new("PointLight")
		light.Name = "CoinLight"
		light.Color = Color3.fromRGB(255, 222, 88)
		light.Brightness = 1.8
		light.Range = 8
		light.Parent = coin
	end

	if not coin:FindFirstChild("CoinHighlight") then
		local highlight = Instance.new("Highlight")
		highlight.Name = "CoinHighlight"
		highlight.Adornee = coin
		highlight.FillColor = Color3.fromRGB(255, 217, 82)
		highlight.OutlineColor = Color3.fromRGB(255, 255, 255)
		highlight.FillTransparency = 0.35
		highlight.OutlineTransparency = 0.05
		highlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
		highlight.Parent = coin
	end
end

local function spawnCoin()
	if activeCoinCount >= MAX_COINS then
		return
	end

	local coin = coinTemplate:Clone()
	coin.Name = "Coin"
	coin.CFrame = getRandomCoinCFrame()
	coin.Anchored = true
	coin.CanCollide = false
	coin.CanTouch = true
	coin.Parent = spawnedCoinsFolder

	applyCoinVisuals(coin)

	activeCoinCount += 1
	animatedCoins[coin] = {
		baseCFrame = coin.CFrame,
		seed = math.random() * 1000,
		spinSpeed = math.rad(math.random(100, 150)),
	}

	local taken = false
	local touchConnection

	touchConnection = coin.Touched:Connect(function(hit)
		if taken then
			return
		end

		local character = hit.Parent
		if not character then
			return
		end

		local humanoid = character:FindFirstChildOfClass("Humanoid")
		if not humanoid then
			return
		end

		local player = Players:GetPlayerFromCharacter(character)
		if not player then
			return
		end

		taken = true
		coin.CanTouch = false

			local coinsObject = getCoinsObject(player)
			if not coinsObject then
				return
			end
			local coinsBefore = coinsObject.Value
			local reward = calculateFinalCoinGain(player, 1)
			local rebirth = getRebirthFolder(player)
			local floorReward = rebirth and syncRebirthCoinMultiplierBonus(player) or 1
			if reward < floorReward then
				reward = floorReward
			end
			addCurrency(coinsObject, reward)
			local coinsAdded = roundToTenth(coinsObject.Value - coinsBefore)
			if math.abs(coinsAdded - reward) > 0.001 then
				warn(string.format("[CoinGainMismatch] %s reward=%s added=%s before=%s after=%s", player.Name, tostring(reward), tostring(coinsAdded), tostring(coinsBefore), tostring(coinsObject.Value)))
			end
			if rebirth and rebirth.SecondAreaUnlocked.Value then
					local xpUpgrades = getXPUpgradesFolder(player)
					local xpPerCoin = 1 + (xpUpgrades and xpUpgrades:FindFirstChild("XPMultiplierLevel") and xpUpgrades.XPMultiplierLevel.Value or 0)
					local _, _, _, passiveXpMul = getPassiveSpecialBoosts(player)
					local _, _, _, runeXpMul = getRuneBonusMultipliers(player)
					local _, _, _, eventXPMul = getServerEventMultipliers()
					local xpReward = roundToTenth(xpPerCoin * getXPBoostMultiplier(player) * passiveXpMul * runeXpMul * eventXPMul)
				addCurrency(getXPCurrencyObject(player), xpReward)
				firePickup(player, "XP", xpReward)
			end
			markDirty(player)

		if touchConnection then
			touchConnection:Disconnect()
		end

		animatedCoins[coin] = nil
		activeCoinCount -= 1
		coin:Destroy()

			firePickup(player, "Coin", coinsAdded)

		local respawnDelay = getCoinRespawnDelay(player)
		task.delay(respawnDelay, function()
			if zonePart.Parent and coinTemplate.Parent then
				spawnCoin()
			end
		end)
	end)
end

local function savePlayerData(player, force)
	if not player or player.Parent ~= Players then
		return false
	end

	if not dirtyPlayers[player] and not force then
		return true
	end

	local now = os.clock()
	local lastSave = lastSaveTimes[player] or 0

	if not force and (now - lastSave) < SAVE_COOLDOWN then
		return false
	end

	local payload = buildSaveData(player)
	if not payload then
		return false
	end

	local success, result = pcall(function()
		playerDataStore:SetAsync("Player_" .. player.UserId, payload)
	end)

	if success then
		lastSaveTimes[player] = now
		dirtyPlayers[player] = nil
		return true
	else
		warn("Не удалось сохранить " .. player.Name .. ": " .. tostring(result))
		return false
	end
end

		return playerDataStore:GetAsync("Player_" .. player.UserId)
	end)

		warn("Не удалось загрузить " .. player.Name .. ": " .. tostring(data))
		return
	end

	if type(data) ~= "table" then
		return
	end

	local coinUpgrades = getCoinUpgradesFolder(player)
	local woodCurrency = getWoodCurrencyObject(player)
	local woodUpgrades = getWoodUpgradesFolder(player)
	local hayCurrency = getHayCurrencyObject(player)

	runeUpgrades.RuneSpeedCost.Value = math.max(1, tonumber(savedRuneUpgrades.RuneSpeedCost) or RUNE_UPGRADES.Speed.startCost)

		end

			cost.Value = math.max(0, tonumber(savedXPUpgrades[cfg.costName]) or fallbackCost)
		end
	end

		local loaded = emptyRuneState()
		local savedNature = type(savedRunes.natureCounts) == "table" and savedRunes.natureCounts or (type(savedRunes.counts) == "table" and savedRunes.counts or {})

		for runeId, _ in pairs(loaded.natureCounts) do
			loaded.natureCounts[runeId] = math.max(0, tonumber(savedNature[runeId]) or 0)
			loaded.counts[runeId] = loaded.natureCounts[runeId]
		end
		for runeId, _ in pairs(loaded.forestCounts) do
			loaded.forestCounts[runeId] = math.max(0, tonumber(savedForest[runeId]) or 0)
		end
		runeStates[player] = loaded
	else
		runeStates[player] = emptyRuneState()
	end
	writeRuneStateValue(player)

	updateUpgradeActiveFlags(coinUpgrades, COIN_UPGRADES)
	updateUpgradeActiveFlags(woodUpgrades, WOOD_UPGRADES)
	updateUpgradeActiveFlags(paperUpgrades, PAPER_UPGRADES)

	if type(data.Position) == "table" and data.Position.X and data.Position.Y and data.Position.Z then
		pendingTeleportTransforms[player] = data.Position
	end
end

local function applyPendingTeleport(player, character)
	local savedPosition = pendingTeleportTransforms[player]
	if not savedPosition then
		return
	end

	task.spawn(function()
		local hrp = character:WaitForChild("HumanoidRootPart", 10)
		if not hrp then
			return
		end

		task.wait(0.25)

		local position = Vector3.new(savedPosition.X, savedPosition.Y, savedPosition.Z)
		local look = Vector3.new(savedPosition.LookX or 0, savedPosition.LookY or 0, savedPosition.LookZ or -1)

		if look.Magnitude < 0.05 then
			look = Vector3.new(0, 0, -1)
		else
			look = look.Unit
		end

		hrp.CFrame = CFrame.lookAt(position, position + look)
		pendingTeleportTransforms[player] = nil
	end)
end

adminActionEvent.OnServerEvent:Connect(function(player, actionName, currencyName, actionValue)
	if player.Name ~= ADMIN_NAME then
		return
	end
	local targetPlayer = player
	if type(actionValue) == "table" and type(actionValue.targetName) == "string" then
		local targetName = string.lower(actionValue.targetName)
		if targetName ~= "" and targetName ~= "me" and targetName ~= "self" then
			for _, candidate in ipairs(Players:GetPlayers()) do
				local nameLower = string.lower(candidate.Name)
				local displayLower = string.lower(candidate.DisplayName or "")
				if nameLower == targetName or displayLower == targetName or string.sub(nameLower, 1, #targetName) == targetName then
					targetPlayer = candidate
					break
				end
			end
			if not targetPlayer or targetPlayer == player and string.lower(player.Name) ~= targetName and string.lower(player.DisplayName or "") ~= targetName then
				fireSimple(player, "Игрок не найден: " .. tostring(actionValue.targetName))
				return
			end
		end
	end

	if actionName == "GiveCurrency" then
		local amount = roundToTenth(type(actionValue) == "table" and actionValue.amount or actionValue)
		if amount <= 0 then
			fireSimple(player, "Введи число больше нуля")
			return
		end

		local currencyObject = getCurrencyObjectForName(targetPlayer, currencyName)
		if not currencyObject then
			fireSimple(player, "Неизвестная валюта")
			return
		end

		addCurrency(currencyObject, amount)
		markDirty(targetPlayer)
		fireSimple(player, "Валюта выдана: " .. targetPlayer.Name)
	elseif actionName == "GivePassive" then
		local passiveId = tostring(currencyName or "")
		if not PASSIVE_DEFS[passiveId] then
			fireSimple(player, "Пассив не найден")
			return
		end
		local state = getPassiveState(targetPlayer)
		if #state.inventory >= PASSIVE_INVENTORY_CAPACITY then
			fireSimple(player, "Инвентарь пассивов заполнен")
			return
		end
		table.insert(state.inventory, {
			uid = HttpService:GenerateGUID(false),
			passiveId = passiveId,
		})
		writePassiveStateValue(targetPlayer)
		markDirty(targetPlayer)
		firePassiveRoll(targetPlayer, passiveId)
		fireSimple(player, "Пассив добавлен: " .. passiveId .. " -> " .. targetPlayer.Name)
	elseif actionName == "AdminRuneBoost" then
		local runeUpgrades = getRuneUpgradeFolder(targetPlayer)
		if runeUpgrades then
			runeUpgrades.RuneLuckLevel.Value = math.max(runeUpgrades.RuneLuckLevel.Value, 20)
			runeUpgrades.RuneSpeedLevel.Value = math.max(runeUpgrades.RuneSpeedLevel.Value, 20)
			runeUpgrades.RuneBulkLevel.Value = math.max(runeUpgrades.RuneBulkLevel.Value, 20)
			markDirty(targetPlayer)
			fireSimple(player, "Rune boosts x20 применены -> " .. targetPlayer.Name)
		end
	elseif actionName == "StartServerEvent" then
		local eventId = tostring(currencyName or "")
		local duration = tonumber(actionValue) or ADMIN_EVENT_DURATION
		if duration <= 0 then duration = ADMIN_EVENT_DURATION end
		local nowTime = os.clock()
		if eventId == "All" then
			for key in pairs(ACTIVE_SERVER_EVENTS) do
				local current = activeAdminEvents[key] and activeAdminEvents[key].endsAt or nowTime
				activeAdminEvents[key] = {endsAt = math.max(current, nowTime) + duration}
			end
		elseif ACTIVE_SERVER_EVENTS[eventId] then
			local current = activeAdminEvents[eventId] and activeAdminEvents[eventId].endsAt or nowTime
			activeAdminEvents[eventId] = {endsAt = math.max(current, nowTime) + duration}
		else
			fireSimple(player, "Неизвестный event id")
			return
		end
		broadcastAdminEventState()
		publishGlobalAdminEvents()
		fireSimple(player, "Server event запущен")
	elseif actionName == "ResetSelf" then
		resetAllPlayerData(player)
		fireSimple(player, "Данные сброшены")
	end
end)

task.spawn(function()
	while true do
		task.wait(0.2)

		local now = os.clock()
		local eventsChanged = false
			for eventId, eventState in pairs(activeAdminEvents) do
				local endAt = type(eventState) == "table" and (eventState.endsAt or 0) or tonumber(eventState) or 0
				if endAt > 0 and endAt <= now then
					activeAdminEvents[eventId] = nil
					eventsChanged = true
				end
			end
		if eventsChanged then
			broadcastAdminEventState()
			publishGlobalAdminEvents()
		end

			for _, player in ipairs(Players:GetPlayers()) do
				local rebirth = getRebirthFolder(player)
				local paperFactory = getPaperFactoryFolder(player)

			if rebirth and paperFactory then
				if not rebirth.FourthSystemsUnlocked.Value then
					paperFactory.IsRunning.Value = false
					paperFactory.Countdown.Value = BASE_PAPER_PRODUCTION_TIME
					paperProcessAnchors[player] = now
				else
					local cycleTime = getPaperCycleTime(player)

					if paperFactory.Fuel.Value > 0 then
						local lastTick = paperProcessAnchors[player] or now
						local elapsed = now - lastTick

						while elapsed >= cycleTime and paperFactory.Fuel.Value > 0 do
							paperFactory.Fuel.Value -= 1
							paperFactory.Paper.Value = roundToTenth(paperFactory.Paper.Value + getPaperProductionAmount(player))
							lastTick += cycleTime
							elapsed = now - lastTick
							markDirty(player)
						end

						paperProcessAnchors[player] = lastTick
						paperFactory.IsRunning.Value = paperFactory.Fuel.Value > 0

						if paperFactory.Fuel.Value > 0 then
							paperFactory.Countdown.Value = roundToTenth(math.max(0, cycleTime - (now - lastTick)))
						else
							paperFactory.Countdown.Value = cycleTime
						end
					else
						paperFactory.IsRunning.Value = false
						paperFactory.Countdown.Value = cycleTime
						paperProcessAnchors[player] = now
					end
					end
					end

					if isCharacterOnHayBlock(player) then
						activeHayCollection[player] = true
						local nextHayAt = hayCooldowns[player.UserId] or 0
						if now >= nextHayAt then
							local hayReward = getHayReward(player)
							if hayReward > 0 then
								addCurrency(getHayCurrencyObject(player), hayReward)
								firePickup(player, "Hay", hayReward)
								markDirty(player)
							end
							hayCooldowns[player.UserId] = now + getHayCooldown(player)
						end
					else
						activeHayCollection[player] = nil
						hayCooldowns[player.UserId] = now
					end

					local function processRuneSet(setName, activeTable)
						local cfg = RUNE_SET_DEFS[setName]
						if not cfg or not cfg.block then
							activeTable[player] = nil
							stopRuneRolling(player, setName)
							return
						end
						local rebirthCount = tonumber(rebirth and rebirth.Count.Value) or 0
						if rebirthCount < cfg.unlockRebirth then
							activeTable[player] = nil
							stopRuneRolling(player, setName)
							return
						end
						if not isCharacterOnBlock(player, cfg.block, Vector3.new(0.8, 3.2, 0.8)) then
							activeTable[player] = nil
							stopRuneRolling(player, setName)
							return
						end
						activeTable[player] = true
						startRuneRolling(player, setName)
						local session = runeRollingSessions[player]
						local interval = getRuneRollInterval(player)
						if session and (now - (session.lastRollAt or 0)) >= interval then
							session.lastRollAt = now
							local currencyObject = cfg.currency == "Paper" and getPaperCurrencyObject(player) or getCoinsObject(player)
							if not currencyObject or safeNumber(currencyObject.Value, 0) < cfg.cost then
								local key = "rune_no_" .. setName
								local cd = runeOpenCooldowns[player.UserId .. key] or 0
								if now >= cd then
									fireSimple(player, cfg.insufficientText)
									runeOpenCooldowns[player.UserId .. key] = now + 2
								end
								return
							end
							if not spendCurrency(currencyObject, cfg.cost) then
								return
							end
							local rolled, state, denoms = rollFromSet(player, setName)
							if rolled then
								for _, runeId in ipairs(rolled) do
									session.sessionCounts[runeId] = (session.sessionCounts[runeId] or 0) + 1
								end
								notifyEvent:FireClient(player, {
									kind = "rune_roll_tick",
									system = setName,
									rolled = rolled,
									session = session.sessionCounts,
									state = state,
									effectiveDenominators = denoms,
								})
							end
						end
					end

					processRuneSet("Forest", activeForestRuneRolls)
					processRuneSet("Nature", activeNatureRuneRolls)
			end
		end
end)

RunService.Heartbeat:Connect(function()
	local now = os.clock()

	for coin, info in pairs(animatedCoins) do
		if coin and coin.Parent then
			local bobOffset = math.sin(now * 2 + info.seed) * 0.35
			local rotationY = now * info.spinSpeed + info.seed

			coin.CFrame = info.baseCFrame
				* CFrame.new(0, bobOffset, 0)
				* CFrame.Angles(0, rotationY, 0)
		else
			animatedCoins[coin] = nil
		end
	end

	if treeBlock and treeBlock.Parent then
		local alpha = (math.sin(now * 2.3) + 1) / 2
		local colorA = Color3.fromRGB(116, 77, 48)
		local colorB = Color3.fromRGB(166, 117, 75)

		treeBlock.Color = colorA:Lerp(colorB, alpha)

		local pulseY = treeBaseSize.Y + math.sin(now * 2.3) * 0.35
		treeBlock.Size = Vector3.new(treeBaseSize.X, pulseY, treeBaseSize.Z)
		treeBlock.CFrame = treeBaseCFrame * CFrame.new(0, (pulseY - treeBaseSize.Y) / 2, 0)

		local light = treeBlock:FindFirstChild("TreeLight")
		if light then
			light.Brightness = 1.6 + alpha * 1.4
			light.Color = Color3.fromRGB(180 + math.floor(alpha * 30), 120 + math.floor(alpha * 20), 70)
		end
	end
end)
