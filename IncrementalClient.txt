local function roundToTenth(value)
	return math.floor((tonumber(value) or 0) * 10 + 0.5) / 10
end

local function formatNumber(value)
	value = roundToTenth(value)

	if math.abs(value - math.floor(value)) < 0.001 then
		return tostring(math.floor(value))
	end

	return string.format("%.1f", value)
end

local function getAssetOrEmpty(assetId)
	if assetId and assetId ~= "" and not string.find(assetId, "PASTE") then
		return assetId
	end
	return ""
end

local function createCorner(parent, radius)
	local corner = Instance.new("UICorner")
	corner.CornerRadius = UDim.new(0, radius or 12)
	corner.Parent = parent
	return corner
end

local function createStroke(parent, color, thickness, transparency)
	local stroke = Instance.new("UIStroke")
	stroke.Color = color or Color3.fromRGB(255, 255, 255)
	stroke.Thickness = thickness or 1
	stroke.Transparency = transparency or 0
	stroke.Parent = parent
	return stroke
end

local function createGradient(parent, color1, color2, rotation)
	local gradient = Instance.new("UIGradient")
	gradient.Color = ColorSequence.new({
		ColorSequenceKeypoint.new(0, color1),
		ColorSequenceKeypoint.new(1, color2),
	})
	gradient.Rotation = rotation or 90
	gradient.Parent = parent
	return gradient
end

local function addScaleHover(button)
	local scale = Instance.new("UIScale")
	scale.Scale = 1
	scale.Parent = button

	button.MouseEnter:Connect(function()
		TweenService:Create(scale, TweenInfo.new(0.12), {Scale = 1.05}):Play()
	end)

	button.MouseLeave:Connect(function()
		TweenService:Create(scale, TweenInfo.new(0.12), {Scale = 1}):Play()
	end)

	button.MouseButton1Down:Connect(function()
		TweenService:Create(scale, TweenInfo.new(0.08), {Scale = 0.96}):Play()
	end)

	button.MouseButton1Up:Connect(function()
		TweenService:Create(scale, TweenInfo.new(0.08), {Scale = 1.05}):Play()
	end)
end

local function getCoinUpgradeMultiplier()
	return 1 + (coinMultiplierLevel.Value * 0.1)
end

local function getCoinReward()
	local baseValue = 1 + coinValueLevel.Value
	local woodCoinBoostMultiplier = 1.1 ^ woodCoinBoostLevel.Value
	return roundToTenth(baseValue * getCoinUpgradeMultiplier() * rebirthCoinMultiplierBonus.Value * woodCoinBoostMultiplier)
end

local function getCoinRespawn()
	local value = 3 - (coinSpawnLevel.Value * 0.5) - rebirthSpawnSpeedBonus.Value
	if value < 0.5 then
		value = 0.5
	end
	return roundToTenth(value)
end

local function computeGoalWoodFactor(coinAmount)
	local threshold = 10000
	local goalsReached = 0
	local factor = 1

	while factor < 10 and coinAmount >= threshold do
		goalsReached += 1
		factor = math.min(1 + (goalsReached * 0.1), 10)
		threshold = math.floor(threshold * 1.25 + 0.5)
	end

	return factor, goalsReached, threshold
end

local function getWoodReward()
	local baseValue = 1 + woodValueLevel.Value
	local baseMultiplier = 1 + woodMultiplierLevel.Value + (coinWoodBoostLevel.Value * 0.25)

	local goalFactor = 1
	if fourthSystemsUnlocked.Value then
		goalFactor = select(1, computeGoalWoodFactor(coins.Value))
	end

	return roundToTenth(baseValue * baseMultiplier * goalFactor)
end

local function getWoodCooldown()
	local value = 3 - (woodSpeedLevel.Value * 0.25)
	if value < 0.5 then
		value = 0.5
	end
	return roundToTenth(value)
end

		Position = holder.Position - UDim2.new(0, 0, 0.04, 0)
	})

	tweenUp1:Play()
	scaleTween:Play()

	tweenUp1.Completed:Connect(function()
		tweenUp2:Play()
		fadeText:Play()
		fadeIcon:Play()
		moveDown:Play()
	end)

	task.delay(1, function()
		holder:Destroy()
	end)
end

local function setButtonVisual(button, text, canBuy, atMax, activeColor, disabledColor, textColor, locked)
	if locked then
		button.BackgroundColor3 = Color3.fromRGB(100, 110, 125)
		button.TextColor3 = Color3.fromRGB(240, 244, 255)
		button.Text = "LOCKED"
		button.AutoButtonColor = false
		return
	end

	if atMax then
		button.BackgroundColor3 = Color3.fromRGB(138, 121, 188)
		button.TextColor3 = Color3.fromRGB(255, 255, 255)
		button.Text = "MAX"
		button.AutoButtonColor = false
		return
	end

	if canBuy then
		button.BackgroundColor3 = activeColor
		button.TextColor3 = textColor
		button.AutoButtonColor = true
	else
		button.BackgroundColor3 = disabledColor
		button.TextColor3 = Color3.fromRGB(242, 247, 255)
		button.AutoButtonColor = false
	end

	button.Text = text
end

local function updateCard(card, levelValue, costValue, activeValue, maxLevel, currentCurrency, lockedReason)
	local locked = lockedReason ~= nil
	local atMax = (not locked) and maxLevel ~= math.huge and levelValue >= maxLevel
	local canBuy = (not locked) and (not atMax) and costValue > 0 and currentCurrency >= costValue

	if maxLevel == math.huge then
		card.levelLabel.Text = "Level " .. tostring(levelValue) .. "/∞"
	else
		card.levelLabel.Text = "Level " .. tostring(levelValue) .. "/" .. tostring(maxLevel)
	end

	if locked then
		card.costLabel.Text = "Cost: LOCKED"
		card.statusLabel.Text = lockedReason
	elseif atMax then
		card.costLabel.Text = "Cost: MAX"
		card.statusLabel.Text = "Status: MAX"
	else
		card.costLabel.Text = "Cost: " .. tostring(costValue)
		card.statusLabel.Text = activeValue and "Status: Active" or "Status: Inactive"
	end

	setButtonVisual(card.buyButton, "BUY!", canBuy, atMax, Color3.fromRGB(99, 243, 200), Color3.fromRGB(136, 148, 175), Color3.fromRGB(22, 48, 62), locked)
	setButtonVisual(card.maxButton, "MAX!", canBuy, atMax, Color3.fromRGB(177, 122, 255), Color3.fromRGB(136, 148, 175), Color3.fromRGB(255, 255, 255), locked)
end

local function updateBoards()
	coinValueLabel.Text = formatNumber(coins.Value)
	coinRewardLabel.Text = formatNumber(getCoinReward())
	coinUpgradeMultiplierLabel.Text = "x" .. formatNumber(getCoinUpgradeMultiplier())
	coinRebirthMultiplierLabel.Text = "x" .. formatNumber(rebirthCoinMultiplierBonus.Value)
	coinRespawnLabel.Text = formatNumber(getCoinRespawn()) .. "s"
	rebirthCountLabel.Text = tostring(rebirthCount.Value)

	updateCard(coinGainCard, coinValueLevel.Value, coinValueCost.Value, coinValueActive.Value, math.huge, coins.Value)
	updateCard(coinMultiCard, coinMultiplierLevel.Value, coinMultiplierCost.Value, coinMultiplierActive.Value, math.huge, coins.Value)
	updateCard(coinSpawnCard, coinSpawnLevel.Value, coinSpawnCost.Value, coinSpawnActive.Value, MAX_COIN_SPAWN_SPEED_LEVEL, coins.Value)
	updateCard(
		coinWoodBoostCard,
		coinWoodBoostLevel.Value,
		coinWoodBoostCost.Value,
		coinWoodBoostActive.Value,
		10,
		coins.Value,
		thirdUpgradeUnlocked.Value and nil or "Откроется на 3-м"
	)

	rebirthLevelText.Text = "Rebirth " .. tostring(rebirthCount.Value) .. "/" .. tostring(MAX_REBIRTH_COUNT)
	rebirthCurrentText.Text = "Сейчас: x" .. formatNumber(rebirthCoinMultiplierBonus.Value) .. " | -" .. formatNumber(rebirthSpawnSpeedBonus.Value) .. "с"

	if rebirthCount.Value == 0 then
		rebirthNextText.Text = "1-е: x2 монеты и -0.1с к спавну"
	elseif rebirthCount.Value == 1 then
		rebirthNextText.Text = "2-е: x5 поверх и открытие дерева"
	elseif rebirthCount.Value == 2 then
		rebirthNextText.Text = "3-е: x10 монет и новые апгрейды"
	elseif rebirthCount.Value == 3 then
		rebirthNextText.Text = "4-е: доска буста и производство бумаги"
	else
		rebirthNextText.Text = "Все перерождения куплены"
	end

	rebirthUnlockText.Text = secondAreaUnlocked.Value and "Новая зона: открыта" or "Новая зона: закрыта"

	local rebirthAtMax = rebirthNextCost.Value <= 0
	local currencyLabel = rebirthNextCurrency.Value ~= "None" and rebirthNextCurrency.Value or ""
	local rebirthCanBuy = false
	if not rebirthAtMax then
		if currencyLabel == "Coins" then
			rebirthCanBuy = coins.Value >= rebirthNextCost.Value
		elseif currencyLabel == "Wood" then
			rebirthCanBuy = woodCurrency.Value >= rebirthNextCost.Value
		end
	end

	if rebirthAtMax then
		rebirthCostText.Text = "Cost: MAX"
	else
		rebirthCostText.Text = "Cost: " .. tostring(rebirthNextCost.Value) .. " " .. currencyLabel
	end

	setButtonVisual(rebirthButton, "REBIRTH!", rebirthCanBuy, rebirthAtMax, Color3.fromRGB(255, 218, 96), Color3.fromRGB(136, 148, 175), Color3.fromRGB(65, 46, 93))

	treeCurrencyText.Text = formatNumber(woodCurrency.Value)
	treeRewardText.Text = formatNumber(getWoodReward())
	treeMultiplierText.Text = "x" .. formatNumber(1 + woodMultiplierLevel.Value + (coinWoodBoostLevel.Value * 0.25))
	treeCooldownText.Text = formatNumber(getWoodCooldown()) .. "s"
	treeUnlockStatusText.Text = secondAreaUnlocked.Value and "Открыто" or "Закрыто"

	updateCard(treeGainCard, woodValueLevel.Value, woodValueCost.Value, woodValueActive.Value, math.huge, woodCurrency.Value)
	updateCard(treeMultiCard, woodMultiplierLevel.Value, woodMultiplierCost.Value, woodMultiplierActive.Value, math.huge, woodCurrency.Value)
	updateCard(treeSpeedCard, woodSpeedLevel.Value, woodSpeedCost.Value, woodSpeedActive.Value, math.huge, woodCurrency.Value)
	updateCard(
		treeCoinBoostCard,
		woodCoinBoostLevel.Value,
		woodCoinBoostCost.Value,
		woodCoinBoostActive.Value,
		20,
		woodCurrency.Value,
		thirdUpgradeUnlocked.Value and nil or "Откроется на 3-м"
	)

	local goalFactor, goalsReached, nextTarget = computeGoalWoodFactor(coins.Value)
	boostFactorText.Text = "x" .. formatNumber(goalFactor)
	boostGoalsText.Text = tostring(goalsReached)
	boostNextTargetText.Text = goalFactor >= 10 and "MAX" or tostring(nextTarget)
	boostCoinText.Text = formatNumber(coins.Value)

	paperText.Text = formatNumber(paperValue.Value)
	fuelText.Text = tostring(fuelValue.Value)
	countdownText.Text = tostring(factoryCountdown.Value) .. "s"
	factoryStatusText.Text = factoryRunning.Value and "Работает" or "Нет топлива"

	treeInfoLockedOverlay.Visible = not secondAreaUnlocked.Value
	treeUpgradeLockedOverlay.Visible = not secondAreaUnlocked.Value
	boostLockedOverlay.Visible = not fourthSystemsUnlocked.Value
	factoryLockedOverlay.Visible = not fourthSystemsUnlocked.Value

	updateWallVisual()
	refreshAdminCurrencyButtons()
end
