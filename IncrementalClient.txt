function roundToTenth(value)
	return math.floor((tonumber(value) or 0) * 10 + 0.5) / 10
end

function formatNumber(value)
	value = tonumber(value) or 0
	local sign = value < 0 and "-" or ""
	local absValue = math.abs(value)
	local suffixes = {"", "K", "M", "B", "T", "Qa", "Qi", "Sx", "SP", "OC", "NO"}

	if absValue < 1000 then
		if absValue % 1 == 0 then
			return sign .. tostring(math.floor(absValue))
		end
		local exact = string.format("%.2f", absValue)
		exact = exact:gsub("(%..-)0+$", "%1"):gsub("%.$", "")
		return sign .. exact
	end

	local idx = 1
	while absValue >= 1000 and idx < #suffixes do
		absValue /= 1000
		idx += 1
	end
	local digits = absValue >= 100 and 0 or (absValue >= 10 and 1 or 2)
	local formatted = string.format("%." .. tostring(digits) .. "f", absValue)
	formatted = formatted:gsub("(%..-)0+$", "%1"):gsub("%.$", "")
	return sign .. formatted .. suffixes[idx]
end

function getPassiveBonusLines(def)
	local lines = {}
	function addMul(label, value)
		if tonumber(value) and math.abs((tonumber(value) or 1) - 1) > 0.0001 then
			table.insert(lines, string.format("• %s x%s", label, formatNumber(value)))
		end
	end
	function addAdd(label, value)
		if tonumber(value) and math.abs(tonumber(value)) > 0.0001 then
			table.insert(lines, string.format("• %s +%s", label, formatNumber(value)))
		end
	end
	addMul("Coins", def.coinMultiplier)
	addMul("Wood", def.woodMultiplier)
	addMul("Global", def.globalMultiplier)
	addMul("Speed", def.speedMultiplier)
	addMul("XP", def.xpMultiplier)
	addMul("Rune Luck", def.runeLuckMultiplier)
	addMul("Rune Speed", def.runeSpeedMultiplier)
	addMul("Rune Bulk", def.runeBulkMultiplier)
	addAdd("Rune Bulk", def.runeBulkAdd)
	if #lines == 0 then
		table.insert(lines, "• Без доп. бонусов")
	end
	return lines
end

function getAssetOrEmpty(assetId)
	if assetId and assetId ~= "" and not string.find(assetId, "PASTE") then

	return ""
end

function createCorner(parent, radius)

function createStroke(parent, color, thickness, transparency)
	local stroke = Instance.new("UIStroke")

	stroke.Thickness = thickness or 1
	stroke.Transparency = transparency or 0
	stroke.Parent = parent
	return stroke

function createGradient(parent, color1, color2, rotation)
	local gradient = Instance.new("UIGradient")
	gradient.Color = ColorSequence.new({
		ColorSequenceKeypoint.new(0, color1),
		ColorSequenceKeypoint.new(1, color2),
	})
	gradient.Rotation = rotation or 90
	gradient.Parent = parent
	return gradient
end

end

function createIcon(parent, imageId, fallbackText, bg1, bg2)
	local holder = Instance.new("Frame")
	holder.Size = UDim2.new(0, 68, 0, 68)
	holder.BackgroundColor3 = bg1
	holder.BorderSizePixel = 0
	holder.Parent = parent
	createCorner(holder, 999)
	createGradient(holder, bg1, bg2, 45)
	createStroke(holder, Color3.fromRGB(255, 255, 255), 2, 0.12)

	local image = Instance.new("ImageLabel")
	image.Size = UDim2.new(1, -16, 1, -16)
	image.Position = UDim2.new(0, 8, 0, 8)
	image.BackgroundTransparency = 1
	image.ScaleType = Enum.ScaleType.Fit
	image.Image = getAssetOrEmpty(imageId)
	image.Parent = holder

	local fallback = Instance.new("TextLabel")
	fallback.Size = UDim2.new(1, 0, 1, 0)
	fallback.BackgroundTransparency = 1
	fallback.Text = fallbackText
	fallback.TextColor3 = Color3.fromRGB(255, 255, 255)
	fallback.Font = Enum.Font.GothamBlack
	fallback.TextScaled = true
	fallback.Parent = holder

	if image.Image ~= "" then
		fallback.Visible = false
	else
		image.Visible = false
	end

	local scale = Instance.new("UIScale")
	scale.Scale = 1
	scale.Parent = holder

	TweenService:Create(scale, TweenInfo.new(1.1, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true), {
		Scale = 1.06
	}):Play()

	return holder
end

function createHorizontalScroller(parent, positionY, height)
	local scrolling = Instance.new("ScrollingFrame")
	scrolling.Position = UDim2.new(0, 0, 0, positionY)
	scrolling.Size = UDim2.new(1, 0, 1, -positionY - height)
	scrolling.CanvasSize = UDim2.new(0, 0, 0, 0)
	scrolling.ScrollBarThickness = 12
	scrolling.ScrollBarImageColor3 = Color3.fromRGB(170, 194, 255)
	scrolling.BackgroundColor3 = Color3.fromRGB(15, 22, 38)
	scrolling.BackgroundTransparency = 0.25
	scrolling.BorderSizePixel = 0
	scrolling.ScrollingDirection = Enum.ScrollingDirection.X
	scrolling.ElasticBehavior = Enum.ElasticBehavior.WhenScrollable
	scrolling.ScrollingEnabled = true
	scrolling.AutomaticCanvasSize = Enum.AutomaticSize.X
	scrolling.HorizontalScrollBarInset = Enum.ScrollBarInset.Always
	scrolling.VerticalScrollBarInset = Enum.ScrollBarInset.Always
	scrolling.Parent = parent

	local layout = Instance.new("UIListLayout")
	layout.FillDirection = Enum.FillDirection.Horizontal
	layout.HorizontalAlignment = Enum.HorizontalAlignment.Left
	layout.VerticalAlignment = Enum.VerticalAlignment.Center
	layout.Padding = UDim.new(0, 12)
	layout.Parent = scrolling

	local padding = Instance.new("UIPadding")
	padding.PaddingLeft = UDim.new(0, 6)
	padding.PaddingRight = UDim.new(0, 6)
	padding.Parent = scrolling

	function updateCanvas()
		scrolling.CanvasSize = UDim2.new(0, layout.AbsoluteContentSize.X + 20, 0, 0)
	end

	layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(updateCanvas)
	updateCanvas()

	return scrolling
end

	if not runeSystemConfig.partIndex then return nil end
	local _, board = createWorldBoard(runeSystemConfig.indexGuiName, runeSystemConfig.partIndex, Enum.NormalId.Front, 72)
	createHeader(board, runeSystemConfig.id .. " RUNE INDEX", "Выбери руну для деталей")
	local list = Instance.new("ScrollingFrame")
	list.Size = UDim2.new(0.45, 0, 1, -66)
	list.Position = UDim2.new(0, 0, 0, 66)
	list.BackgroundTransparency = 1
	list.BorderSizePixel = 0
	list.ScrollBarThickness = 8
	list.Parent = board
	local layout = Instance.new("UIListLayout")
	layout.Padding = UDim.new(0, 6)
	layout.Parent = list
	layout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
		list.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y + 8)
	end)
	local details = Instance.new("TextLabel")
	details.Size = UDim2.new(0.55, -6, 1, -66)
	details.Position = UDim2.new(0.45, 6, 0, 66)
	details.BackgroundColor3 = Color3.fromRGB(34, 55, 82)
	details.TextColor3 = Color3.fromRGB(236, 245, 255)
	details.Font = Enum.Font.GothamSemibold
	details.TextSize = 14
	details.TextWrapped = true
	details.TextXAlignment = Enum.TextXAlignment.Left
	details.TextYAlignment = Enum.TextYAlignment.Top
	details.BorderSizePixel = 0
	details.Text = "Выбери руну слева"
	details.Parent = board
	createCorner(details, 12)

	local rows = {}
	local function setDetail(runeId)
		local info = RUNE_EFFECT_INFO[runeId] or RUNE_EFFECT_INFO.Grass
		local stat = rows[runeId] and rows[runeId]:GetAttribute("RuntimeText") or ""
		details.Text = (runeSystemConfig.names[runeId] or runeId) .. "\n\n" .. stat .. "\n\nЗа 1 руну: " .. tostring(info.per) .. "\nCap: x" .. formatNumber(info.cap or 1)
	end
	for _, runeId in ipairs(runeSystemConfig.displayOrder) do
		local row = Instance.new("TextButton")
		row.Size = UDim2.new(1, 0, 0, 76)
		row.BackgroundColor3 = Color3.fromRGB(43, 67, 96)
		row.BorderSizePixel = 0
		row.Text = ""
		row.Parent = list
		createCorner(row, 12)
		local icon = Instance.new("ImageLabel")
		icon.Size = UDim2.new(0, 44, 0, 44)
		icon.Position = UDim2.new(0, 6, 0, 16)
		icon.BackgroundTransparency = 1
		icon.Image = getAssetOrEmpty(AssetSlots.RuneIcons[runeId] or ICON_ASSET_IDS.RuneCommon)
		icon.Parent = row
		local text = Instance.new("TextLabel")
		text.Size = UDim2.new(1, -58, 1, -8)
		text.Position = UDim2.new(0, 54, 0, 4)
		text.BackgroundTransparency = 1
		text.TextXAlignment = Enum.TextXAlignment.Left
		text.TextYAlignment = Enum.TextYAlignment.Top
		text.TextWrapped = true
		text.Font = Enum.Font.GothamBold
			text.TextSize = 15
		text.TextColor3 = Color3.fromRGB(236, 245, 255)
		text.Text = runeSystemConfig.names[runeId] or runeId
		text.Parent = row
		rows[runeId] = text
		row.MouseButton1Click:Connect(function()
			setDetail(runeId)

function getRarityColor(rarity)
	if rarity == "Mythic" then return Color3.fromRGB(255, 112, 184) end
	if rarity == "Legendary" then return Color3.fromRGB(255, 197, 92) end
	if rarity == "Epic" then return Color3.fromRGB(204, 133, 255) end
	if rarity == "Rare" then return Color3.fromRGB(97, 171, 255) end
	if rarity == "Uncommon" then return Color3.fromRGB(124, 210, 139) end
	return Color3.fromRGB(198, 212, 238)
end

function getPassiveStyle(passiveId, rarity)
	local seed = 0
	for i = 1, #tostring(passiveId or "") do
		seed += string.byte(passiveId, i)
	end
	local base = getRarityColor(rarity)
	local tint = Color3.fromRGB(70 + (seed % 80), 90 + (seed * 3 % 80), 120 + (seed * 5 % 80))
	return base:Lerp(tint, 0.45), base, tint
end

function openPassiveIndexCard(passiveId)
	if not passiveUi or not passiveUi.indexDetail then
		return
	end
	local def = PASSIVE_DEFS[passiveId]
	if not def then
		return
	end
	local icon = passiveUi.indexDetail:FindFirstChild("CardIcon")
	if not icon then
		icon = Instance.new("ImageLabel")
		icon.Name = "CardIcon"
		icon.Size = UDim2.new(0, 84, 0, 84)
		icon.AnchorPoint = Vector2.new(0.5, 0)
		icon.Position = UDim2.new(0.5, 0, 0, 10)
		icon.BackgroundTransparency = 1
		icon.Parent = passiveUi.indexDetail
	end
	icon.Image = getAssetOrEmpty(AssetSlots.PassiveCardImages[passiveId] or ICON_ASSET_IDS.PassiveCardDefault)
	local detailText = passiveUi.indexDetail:FindFirstChild("Text")
	if detailText and detailText:IsA("TextLabel") then
		local bonusLines = getPassiveBonusLines(def)
		detailText.Position = UDim2.new(0, 10, 0, 100)
		detailText.Size = UDim2.new(1, -20, 1, -148)
		detailText.Text = table.concat({
			def.name .. " [" .. tostring(def.rarity) .. "]",
			"Шанс: " .. tostring(def.chance or "?"),
			"",
			"Бонусы:",
			table.concat(bonusLines, "\n"),
		}, "\n")
	end
	passiveUi.indexDetail.Visible = true
end

		openPassiveIndexCard(passiveId)

function parsePassiveState()
	local ok, decoded = pcall(function()
		return HttpService:JSONDecode(passiveStateJson.Value)
	end)
	if ok and type(decoded) == "table" and type(decoded.inventory) == "table" then
		return decoded
	end
	return {inventory = {}, equippedUid = ""}
end

parseRuneState = function()
	local ok, decoded = pcall(function()
		return HttpService:JSONDecode(runeStateJson.Value)
	end)
	if ok and type(decoded) == "table" then
		if type(decoded.natureCounts) == "table" then
			decoded.counts = decoded.natureCounts
		end
		if type(decoded.counts) == "table" then
			return decoded
		end
	end
	return {counts = {Grass = 0, DarkGrass = 0, Dandelion = 0, Flower = 0, Violet = 0, Rose = 0}}
end

function getEquippedPassiveDef()
	local state = parsePassiveState()
	for _, entry in ipairs(state.inventory) do
		if entry.uid == state.equippedUid then
			return PASSIVE_DEFS[entry.passiveId]
		end
	end
	return nil
end

function getCoinUpgradeMultiplier()
	return 1 + (coinMultiplierLevel.Value * 0.1)
end

function getCoinReward()
	local passive = getEquippedPassiveDef()
	local passiveCoin = passive and (passive.coinMultiplier or 1) or 1
	local passiveGlobal = passive and (passive.globalMultiplier or 1) or 1
	local baseValue = 1 + coinValueLevel.Value
	local woodCoinBoostMultiplier = 1.1 ^ woodCoinBoostLevel.Value
	return roundToTenth(baseValue * getCoinUpgradeMultiplier() * rebirthCoinMultiplierBonus.Value * woodCoinBoostMultiplier * passiveCoin * passiveGlobal)
end

function getCoinRespawn()
	local passive = getEquippedPassiveDef()
	local passiveSpeed = passive and (passive.speedMultiplier or 1) or 1
	local value = 3 - (coinSpawnLevel.Value * 0.5) - rebirthSpawnSpeedBonus.Value
	value /= math.max(1, passiveSpeed)
	if value < 0.5 then
		value = 0.5
	end
	return roundToTenth(value)
end

function computeGoalWoodFactor(coinAmount)
	local threshold = 10000
	local goalsReached = 0
	local factor = 1

	while factor < 10 and coinAmount >= threshold do
		goalsReached += 1
		factor = math.min(1 + (goalsReached * 0.1), 10)
		threshold = math.floor(threshold * 1.25 + 0.5)
	end

	return factor, goalsReached, threshold
end

function getWoodReward()
	local passive = getEquippedPassiveDef()
	local passiveWood = passive and (passive.woodMultiplier or 1) or 1
	local passiveGlobal = passive and (passive.globalMultiplier or 1) or 1
	local baseValue = 1 + woodValueLevel.Value
	local baseMultiplier = 1 + woodMultiplierLevel.Value + (coinWoodBoostLevel.Value * 0.25)

	local goalFactor = 1
	if fourthSystemsUnlocked.Value then
		goalFactor = select(1, computeGoalWoodFactor(coins.Value))
	end

	return roundToTenth(baseValue * baseMultiplier * goalFactor * rebirthWoodMultiplierBonus.Value * passiveWood * passiveGlobal)
end

function getWoodCooldown()
	local passive = getEquippedPassiveDef()
	local passiveSpeed = passive and (passive.speedMultiplier or 1) or 1
	local value = 3 - (woodSpeedLevel.Value * 0.25)
	if value < 0.5 then
		value = 0.5
	end
	return roundToTenth(value / math.max(1, passiveSpeed))
end

function getRuneSpeedOverflowBulk(speedStat)
	local interval = 1 / math.max(1, (1 + (math.max(0, speedStat) * 0.5)))
	if interval >= 0.01 then
		return 0
	end
	local overflowFactor = math.min(250, 0.01 / math.max(0.000001, interval))
	return math.max(0, math.floor((overflowFactor - 1) * 2 + 0.5))
end

function getRuneRollIntervalFromSpeed(speedStat)
	local baseInterval = 1 / math.max(1, (1 + (math.max(0, speedStat) * 0.5)))
	return math.max(0.01, baseInterval)
end

function getClientEffectiveRuneStats()
	local baseLuck = 1 + runeLuckLevel.Value + runeLuckXpLevel.Value
	local baseSpeed = 1 + runeSpeedLevel.Value + runeSpeedXpLevel.Value
	local baseBulkBonus = runeBulkLevel.Value + runeBulkXpLevel.Value
	local runeState = parseRuneState()
	local runeCounts = runeState.counts or {}
	local mythicCount = tonumber(runeCounts.Mythic) or 0
	local mythicLuckSpeedMul = math.min(5, 1 + (mythicCount * 0.1))
	local mythicBulkAdd = math.min(25, math.floor(mythicCount * 0.5 + 0.5))
	local passive = getEquippedPassiveDef()
	local passiveLuckMul = passive and (passive.runeLuckMultiplier or 1) or 1
	local passiveSpeedMul = passive and (passive.runeSpeedMultiplier or 1) or 1
	local passiveBulkMul = passive and (passive.runeBulkMultiplier or 1) or 1
	local passiveGlobalMul = passive and (passive.globalMultiplier or 1) or 1
	local passiveCommonSpeedMul = passive and (passive.speedMultiplier or 1) or 1
	local eventLuckMul, eventSpeedMul, eventBulkMul, eventBulkAdd = getActiveAdminEventUpgrades()
	local luck = math.floor(baseLuck * passiveLuckMul * passiveGlobalMul * mythicLuckSpeedMul * eventLuckMul + 0.5)
	local speed = math.floor(baseSpeed * passiveSpeedMul * passiveGlobalMul * passiveCommonSpeedMul * mythicLuckSpeedMul * eventSpeedMul + 0.5)
	local bulkBonus = math.floor((baseBulkBonus + mythicBulkAdd) * passiveBulkMul * passiveGlobalMul + 0.5)
	bulkBonus = math.floor(bulkBonus * eventBulkMul + 0.5) + eventBulkAdd
	if rebirthCount.Value >= 7 then
		luck = math.floor(luck * 1.5 + 0.5)
	end
	bulkBonus += getRuneSpeedOverflowBulk(speed)
	local bulk = math.max(1, 1 + bulkBonus)
	return luck, speed, bulk
end

getActiveAdminEventUpgrades = function()
	local luckMul, speedMul, bulkMul, bulkAdd = 1, 1, 1, 0
	for _, ev in ipairs(activeAdminEvents or {}) do
		if (tonumber(ev.timeLeft) or 0) > 0 then
			local cfg = ADMIN_EVENT_UPGRADES[tostring(ev.key or "")]
			if cfg then
				luckMul *= cfg.runeLuckMul or 1
				speedMul *= cfg.runeSpeedMul or 1
				bulkMul *= cfg.runeBulkMul or 1
				bulkAdd += cfg.runeBulkAdd or 0
			end
		end
	end
	return luckMul, speedMul, bulkMul, bulkAdd
end

function formatAdminEventTime(timeLeft)
	local total = math.max(0, math.floor(tonumber(timeLeft) or 0))
	if total < 60 then
		return tostring(total) .. "с"
	end
	local minutes = math.floor(total / 60)
	local seconds = total % 60
	if minutes < 60 then
		return string.format("%dм %dс", minutes, seconds)
	end
	local hours = math.floor(minutes / 60)
	local remMinutes = minutes % 60
	return string.format("%dч %dм", hours, remMinutes)
end

function normalizeAdminEvents(rawEvents)
	local normalized = {}
	if type(rawEvents) ~= "table" then
		return normalized
	end
	if #rawEvents > 0 then
		for idx = 1, #rawEvents do
			local ev = rawEvents[idx]
			if type(ev) == "table" then
				local key = tostring(ev.key or ev.eventId or "")
				local left = tonumber(ev.timeLeft or ev.remaining or ev.left) or 0
				if key ~= "" and left > 0 then
					normalized[#normalized + 1] = {
						key = key,
						timeLeft = left,
						tooltip = adminEventDescs[key] or tostring(ev.tooltip or ""),
					}
				end
			end
		end
	else
		for key, remain in pairs(rawEvents) do
			local left = tonumber(remain) or 0
			local keyText = tostring(key)
			if keyText ~= "" and left > 0 then
				normalized[#normalized + 1] = {
					key = keyText,
					timeLeft = left,
					tooltip = adminEventDescs[keyText] or "",
				}
			end
		end
	end
	return normalized
end

	dimmer.BackgroundTransparency = 1
	dimmer.ZIndex = 91
	dimmer.Parent = passiveUi.rollFxLayer
	TweenService:Create(dimmer, TweenInfo.new(0.8), {BackgroundTransparency = 0.28}):Play()

	local coinLayer = Instance.new("Frame")
	coinLayer.Size = UDim2.new(1, 0, 1, 0)
	coinLayer.BackgroundTransparency = 1
	coinLayer.ZIndex = 93
	coinLayer.Parent = passiveUi.rollFxLayer

	local coins = {}
	task.spawn(function()
		local elapsed = 0
		local rewardFrame = nil
		local closeRequested = false
		while elapsed < ROLL_ANIMATION_CONFIG.duration do
			local progress = math.clamp(elapsed / ROLL_ANIMATION_CONFIG.duration, 0, 1)
			table.insert(coins, spawnCoin(coinLayer, progress))
			if elapsed >= ROLL_ANIMATION_CONFIG.rewardRevealTime and not rewardFrame then
				rewardFrame = showReward(passiveUi.rollFxLayer, passiveName, passiveIcon, rarityColor, function()
					closeRequested = true
				end)
			end
			if closeRequested then
				break
			end
			elapsed += ROLL_ANIMATION_CONFIG.coinSpawnInterval
			task.wait(ROLL_ANIMATION_CONFIG.coinSpawnInterval)
		end
		while #coins < ROLL_ANIMATION_CONFIG.totalCoins do
			table.insert(coins, spawnCoin(coinLayer, 1))
		end
		cleanupAnimation(passiveUi.rollFxLayer, dimmer, coins, rewardFrame)
	end)
end

function updateCard(card, levelValue, costValue, activeValue, maxLevel, currentCurrency, lockedReason)
	local locked = lockedReason ~= nil
	local atMax = (not locked) and maxLevel ~= math.huge and levelValue >= maxLevel
	local canBuy = (not locked) and (not atMax) and costValue > 0 and currentCurrency >= costValue

	if maxLevel == math.huge then
		card.levelLabel.Text = "Level " .. tostring(levelValue) .. "/∞"
	else
		card.levelLabel.Text = "Level " .. tostring(levelValue) .. "/" .. tostring(maxLevel)
	end

	if locked then
		card.costLabel.Text = "Cost: LOCKED"
		card.statusLabel.Text = lockedReason
	elseif atMax then
		card.costLabel.Text = "Cost: MAX"
		card.statusLabel.Text = "Status: MAX"
	else
		card.costLabel.Text = "Cost: " .. formatNumber(costValue)
		card.statusLabel.Text = activeValue and "Status: Active" or "Status: Inactive"
	end

	setButtonVisual(card.buyButton, "BUY!", canBuy, atMax, Color3.fromRGB(99, 243, 200), Color3.fromRGB(136, 148, 175), Color3.fromRGB(22, 48, 62), locked)
	setButtonVisual(card.maxButton, "MAX!", canBuy, atMax, Color3.fromRGB(177, 122, 255), Color3.fromRGB(136, 148, 175), Color3.fromRGB(255, 255, 255), locked)
end

function getPaperCycleTime()
	local value = 10 - (paperSpeedLevel.Value * 0.5)
	if value < 5 then
		value = 5
	end
	return roundToTenth(value)
end

function getPaperProductionAmount()
	local baseValue = 1 + paperValueLevel.Value
	local multiplier = 1 + (paperMultiplierLevel.Value * 0.1)
	return roundToTenth(baseValue * multiplier)
end

function isSystemUnlocked(stage)
	if stage == 2 then return secondAreaUnlocked.Value or rebirthCount.Value >= 2 end
	if stage == 3 then return thirdUpgradeUnlocked.Value or rebirthCount.Value >= 3 end
	if stage == 4 then return fourthSystemsUnlocked.Value or rebirthCount.Value >= 4 end
	if stage == 5 then return fifthSystemsUnlocked.Value or rebirthCount.Value >= 5 end
	if stage == 6 then return sixthSystemsUnlocked.Value or rebirthCount.Value >= 6 end
	return false
end

function triggerRuneBurst(systemName)
	local target = runeRollBlock
	if systemName == "Forest" and forestRuneBlock and forestRuneBlock:IsA("BasePart") then
		target = forestRuneBlock
	elseif systemName == "Nature" and natureRuneBlock and natureRuneBlock:IsA("BasePart") then
		target = natureRuneBlock
	end
	if not target or not target:IsA("BasePart") then
		return
	end
	local burst = Instance.new("ParticleEmitter")
	burst.Texture = getAssetOrEmpty(AssetSlots.RuneIcons.Grass)
	if systemName == "Forest" then
		burst.Color = ColorSequence.new(Color3.fromRGB(176, 121, 76), Color3.fromRGB(97, 73, 45))
	else
		burst.Color = ColorSequence.new(Color3.fromRGB(122, 255, 116), Color3.fromRGB(66, 184, 82))
	end
	burst.Speed = NumberRange.new(4, 7)
	burst.Lifetime = NumberRange.new(0.2, 0.35)
	burst.Rate = 0
	burst.SpreadAngle = Vector2.new(360, 360)
	burst.Size = NumberSequence.new({
		NumberSequenceKeypoint.new(0, 0.35),
		NumberSequenceKeypoint.new(1, 0),
	})
	burst.Parent = target
	burst:Emit(14)
	task.delay(0.4, function()
		burst:Destroy()
	end)
end

function getButtonPurchaseContext(button)
	return {
		button = button,
		text = button and button.Text or "",
		active = button and button.Active or false,
		selectable = button and button.Selectable or false,
	}
end

function checkButtonExistsRequirement(context)
	if context.button then
		return true, "ok"
	end
	return false, "missing_button"
end

function checkButtonInteractableRequirement(context)
	if context.active and context.selectable then
		return true, "ok"
	end
	return false, "inactive"
end

function checkButtonDisplayStateRequirement(context)
	if context.text == "LOCKED" then
		return false, "locked"
	end
	if context.text == "MAX" then
		return false, "maxed"
	end
	return true, "ok"
end

function buildButtonPurchaseState(context)
	local existsOk, existsReason = checkButtonExistsRequirement(context)
	if not existsOk then
		return { purchasable = false, reason = existsReason, displayState = "hidden" }
	end

	local interactableOk, interactableReason = checkButtonInteractableRequirement(context)
	if not interactableOk then
		return { purchasable = false, reason = interactableReason, displayState = "disabled" }
	end

	local displayOk, displayReason = checkButtonDisplayStateRequirement(context)
	if not displayOk then
		return { purchasable = false, reason = displayReason, displayState = displayReason }
	end

	return { purchasable = true, reason = "ok", displayState = "active" }
end

function isButtonPurchasable(button)
	local purchaseState = buildButtonPurchaseState(getButtonPurchaseContext(button))
	return purchaseState.purchasable
end

function registerNotifyEventListeners()
	notifyEvent.OnClientEvent:Connect(function(payload)
		if typeof(payload) ~= "table" then
			showToast(tostring(payload))
			return
		end

		if payload.kind == "pickup" then
			local ok = pcall(function()
				showPickup(payload.icon or "Coin", payload.amount or 0)
			end)
			if tostring(payload.icon or "") == "Hay" then
				triggerHayBurst()
			end
			if not ok then
				showToast("+" .. tostring(payload.amount or 0))
			end
		elseif payload.kind == "passive_roll_result" then
			local def = PASSIVE_DEFS[payload.passiveId or ""]
			if def then
				playRovAnimation(def)
				showToast("Новый пассив: " .. def.name .. " [" .. def.rarity .. "]")
			else
				showToast("Пассив получен")
			end
		elseif payload.kind == "rune_roll_result" then
			local rolled = payload.rolled or {}
			showToast("Открыто рун: " .. tostring(#rolled))
		elseif payload.kind == "rune_roll_state" then
			if payload.system == "Forest" then
				if payload.active then
					for k in pairs(forestRuneSessionCounts) do forestRuneSessionCounts[k] = 0 end
					if forestRuneSessionPanel then forestRuneSessionPanel.Visible = true end
				else
					if forestRuneSessionPanel then forestRuneSessionPanel.Visible = false end
				end
			else
				if payload.active then startRuneRolling(payload.system) else stopRuneRolling(payload.system) end
			end
		elseif payload.kind == "rune_roll_tick" then
			if payload.system and payload.system ~= "Nature" then
				return
			end
			local session = payload.session or {}
			effectiveRuneDenominators = payload.effectiveDenominators or effectiveRuneDenominators
			if payload.system == "Forest" then
				for runeId, amount in pairs(session) do forestRuneSessionCounts[runeId] = tonumber(amount) or 0 end
				for runeId, label in pairs(forestRuneSessionRows) do
					label.Text = (RUNE_SYSTEM_UI_CONFIG.Forest.names[runeId] or runeId) .. ": +" .. formatNumber(forestRuneSessionCounts[runeId] or 0)
				end
				if forestRuneSessionPanel then forestRuneSessionPanel.Visible = true end
			else
				for runeId, amount in pairs(session) do runeSessionCounts[runeId] = tonumber(amount) or 0 end
				if not runeRollingActive then startRuneRolling(payload.system) end
				updateRuneSessionGui()
				updateRuneSummaryGui()
				updateRuneWorldDisplay()
			end
			triggerRuneBurst(payload.system)
		elseif payload.kind == "admin_event_state" then
			activeAdminEvents = normalizeAdminEvents(payload.events)
			renderAdminEvents(activeAdminEvents)
			if updateBoards then updateBoards() end
		else
			showToast(payload.text or "Готово")
		end
	end)
end

function registerWorldInputBindings()
	mouse.Button1Down:Connect(function()
		if UserInputService:GetFocusedTextBox() then return end

		if not isSystemUnlocked(2) then
			if passiveUi.indexPanel and mouse.Target == passiveRollBlock then
				passiveUi.indexPanel.Visible = not passiveUi.indexPanel.Visible
			elseif passiveUi.inventoryPanel and mouse.Target == passiveInventoryBlock then
				passiveUi.inventoryPanel.Visible = not passiveUi.inventoryPanel.Visible
			end
			return
		end

		if mouse.Target == treeBlock then
			woodClickEvent:FireServer()
			return
		end

		if passiveUi.indexPanel and mouse.Target == passiveRollBlock then
			if isSystemUnlocked(5) then
				passiveUi.inventoryPanel.Visible = false
				passiveUi.indexPanel.Visible = not passiveUi.indexPanel.Visible
			else
				showToast("Пассивы откроются на 5-м перерождении")
			end
			return
		end

		if passiveUi.inventoryPanel and mouse.Target == passiveInventoryBlock then
			if isSystemUnlocked(5) then
				passiveUi.indexPanel.Visible = false
				passiveUi.inventoryPanel.Visible = not passiveUi.inventoryPanel.Visible
			else
				showToast("Пассивы откроются на 5-м перерождении")
			end
			return
		end

		if mouse.Target == runeRollBlock or mouse.Target == runeIndexBlock then
			if not isSystemUnlocked(6) then
				showToast("Руны откроются на 6-м перерождении")
			end
			return
		end
	end)
end

function startRuntimeUiUpdates()
	if passiveUi.hintEmoji then
		RunService.RenderStepped:Connect(function()
			local t = os.clock()
			local blink = math.sin(t * 1.2)
			if blink > 0.985 then
				passiveUi.hintEmoji.Text = ";)"
			elseif blink < -0.985 then
				passiveUi.hintEmoji.Text = "-_-"
			else
				passiveUi.hintEmoji.Text = "😊"
			end
			passiveUi.hintEmoji.Rotation = math.sin(t * 0.8) * 6
			if runeRollBlock and runeRollBlock:IsA("BasePart") then
				local pulse = (math.sin(t * 1.6) + 1) * 0.5
				local green = math.floor(120 + pulse * 90)
				runeRollBlock.Color = Color3.fromRGB(40 + math.floor(pulse * 24), green, 52 + math.floor(pulse * 24))
				local light = runeRollBlock:FindFirstChild("PassiveFxLight")
				if light and light:IsA("PointLight") then
					light.Brightness = 1.5 + pulse * 1.2
					light.Color = Color3.fromRGB(122, 255, 138)
				end
			end
		end)
	end

	function tickAdminEventTimers()
		if #activeAdminEvents == 0 then return end
		local nextEvents = {}
		for idx = 1, #activeAdminEvents do
			local ev = activeAdminEvents[idx]
			local left = math.max(0, (tonumber(ev.timeLeft) or 0) - 1)
			ev.timeLeft = left
			if left > 0 then nextEvents[#nextEvents + 1] = ev end
		end
		activeAdminEvents = nextEvents
		renderAdminEvents(activeAdminEvents)
	end

	task.spawn(function()
		while true do
			task.wait(1)
			tickAdminEventTimers()
		end
	end)

	task.spawn(function()
		while true do
			task.wait(0.2)
			updateBoards()
		end
	end)

	if #passiveEffectParts > 0 then
		RunService.RenderStepped:Connect(function()
			local t = os.clock()
			for idx, part in ipairs(passiveEffectParts) do
				local alpha = (math.sin(t * (1.3 + idx * 0.2)) + 1) / 2
				part.Color = Color3.fromRGB(95 + math.floor(alpha * 90), 130 + math.floor(alpha * 70), 210 + math.floor(alpha * 45))
				local light = part:FindFirstChild("PassiveFxLight")
				if light then light.Brightness = 1.2 + alpha * 2.1 end
			end
		end)
	end
end
